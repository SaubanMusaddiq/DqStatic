12. ಯೂಸುಫ್
ಪರಮ ದಯಾಮಯನೂ ಕರುಣಾಳುವೂ ಆದ ಅಲ್ಲಾಹನ ನಾಮದಿಂದ
12:1 - 3
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ الٓر ۚ تِلْكَ ءَايَـٰتُ ٱلْكِتَـٰبِ ٱلْمُبِينِ١إِنَّآ أَنزَلْنَـٰهُ قُرْءَٰنًا عَرَبِيًّا لَّعَلَّكُمْ تَعْقِلُونَ٢نَحْنُ نَقُصُّ عَلَيْكَ أَحْسَنَ ٱلْقَصَصِ بِمَآ أَوْحَيْنَآ إِلَيْكَ هَـٰذَا ٱلْقُرْءَانَ وَإِن كُنتَ مِن قَبْلِهِۦ لَمِنَ ٱلْغَـٰفِلِينَ٣
ಅಲಿಫ್ ಲಾಮ್ ರಾ. ಇವು ತನ್ನ ಅಭಿಪ್ರಾಯವನ್ನು ಸುವ್ಯಕ್ತವಾಗಿ ವಿವರಿಸಿ ಹೇಳುವ ಗ್ರಂಥದ ಸೂಕ್ತಗಳು. ಇದನ್ನು ಚೆನ್ನಾಗಿ ತಿಳಿದುಕೊಳ್ಳಲು ಸಾಧ್ಯವಾಗುವಂತೆ1 ನಾವು ಇದನ್ನು ಕುರ್‌ಆನನ್ನಾಗಿ2 ಮಾಡಿ ಅರಬೀ ಭಾಷೆಯಲ್ಲಿ ಅವತೀರ್ಣಗೊಳಿಸಿದೆವು. ಓ ಪೈಗಂಬರರೇ, ನಾವು ಈ ಕುರ್‌ಆನನ್ನು ನಿಮ್ಮ ಕಡೆಗೆ 'ದಿವ್ಯವಾಣಿ' ಮಾಡಿ, ವೃತ್ತಾಂತಗಳನ್ನೂ ವಸ್ತುಸ್ಥಿತಿಗಳನ್ನೂ ಉತ್ತಮ ಶೈಲಿಯಲ್ಲಿ ನಿಮಗೆ ವಿವರಿಸಿಕೊಡುತ್ತೇವೆ. ಅನ್ಯಥಾ ಇದಕ್ಕಿಂತ ಮುಂಚೆ ನೀವು (ಈ ವಿಷಯಗಳ ಕುರಿತು) ಸಂಪೂರ್ಣ ಅಜ್ಞರಾಗಿದ್ದಿರಿ.3
12:4 - 6
إِذْ قَالَ يُوسُفُ لِأَبِيهِ يَـٰٓأَبَتِ إِنِّى رَأَيْتُ أَحَدَ عَشَرَ كَوْكَبًا وَٱلشَّمْسَ وَٱلْقَمَرَ رَأَيْتُهُمْ لِى سَـٰجِدِينَ٤قَالَ يَـٰبُنَىَّ لَا تَقْصُصْ رُءْيَاكَ عَلَىٰٓ إِخْوَتِكَ فَيَكِيدُوا۟ لَكَ كَيْدًا ۖ إِنَّ ٱلشَّيْطَـٰنَ لِلْإِنسَـٰنِ عَدُوٌّ مُّبِينٌ٥وَكَذَٰلِكَ يَجْتَبِيكَ رَبُّكَ وَيُعَلِّمُكَ مِن تَأْوِيلِ ٱلْأَحَادِيثِ وَيُتِمُّ نِعْمَتَهُۥ عَلَيْكَ وَعَلَىٰٓ ءَالِ يَعْقُوبَ كَمَآ أَتَمَّهَا عَلَىٰٓ أَبَوَيْكَ مِن قَبْلُ إِبْرَٰهِيمَ وَإِسْحَـٰقَ ۚ إِنَّ رَبَّكَ عَلِيمٌ حَكِيمٌ٦
ಇದು ಯೂಸುಫರು ತಮ್ಮ ಪಿತರೊಡನೆ ಹೇಳಿದ ಸಂದರ್ಭ- "ಅಪ್ಪಾ, ಹನ್ನೊಂದು ನಕ್ಷತ್ರಗಳೂ ಸೂರ್ಯನೂ ಚಂದ್ರನೂ ನನಗೆ ಸಾಷ್ಟಾಂಗವೆರಗುತ್ತಿರುವುದನ್ನು ನಾನು ಸ್ವಪ್ನದಲ್ಲಿ ಕಂಡಿರುತ್ತೇನೆ." ಉತ್ತರವಾಗಿ ಅವರ ತಂದೆ ಹೀಗೆಂದರು, "ಮಗನೇ, ನಿನ್ನ ಈ ಸ್ವಪ್ನವನ್ನು ನಿನ್ನ ಸಹೋದರರೊಡನೆ ಹೇಳಬೇಡ. ಅನ್ಯಥಾ ಅವರು ನಿನ್ನ ವಿರುದ್ಧ ಒಳಸಂಚು ನಡೆಸುವರು.4 ವಾಸ್ತವದಲ್ಲಿ ಶೈತಾನನು ಮಾನವನ ಪ್ರತ್ಯಕ್ಷ ಶತ್ರು. (ನೀನು ಸ್ವಪ್ನದಲ್ಲಿ ಕಂಡ) ಹಾಗೆಯೇ ಆಗುವುದು. ನಿನ್ನ ಪ್ರಭುವು ನಿನ್ನನ್ನು (ತನ್ನ ಕಾರ್ಯಕ್ಕಾಗಿ) ಆಯ್ದುಕೊಳ್ಳುವನು.5 ನಿನಗೆ ವಿಷಯಗಳ ಆಳಕ್ಕೆ ತಲುಪುವುದನ್ನು ಕಲಿಸಿಕೊಡುವನು6 ಮತ್ತು ಇದಕ್ಕೆ ಮುಂಚೆ, ನಿನ್ನ ಪೂರ್ವಜರಾಗಿದ್ದ ಇಬ್ರಾಹೀಮ್ ಮತ್ತು ಇಸ್‌ಹಾಕ್‌ರ ಮೇಲೆ ಅವನು ತನ್ನ 'ಭವ್ಯಕೊಡುಗೆ'ಯನ್ನು ಪೂರ್ಣಗೊಳಿಸಿದಂತೆ ನಿನ್ನ ಮೇಲೂ ಯಾ‌ಕೂಬರ ವಂಶಜರ ಮೇಲೂ ಪೂರ್ಣಗೊಳಿಸುವನು. ನಿಶ್ಚಯವಾಗಿಯೂ ನಿನ್ನ ಪ್ರಭು ಸರ್ವಜ್ಞನೂ ಧೀಮಂತನೂ ಆಗಿರುತ್ತಾನೆ."7

۞ لَّقَدْ كَانَ فِى يُوسُفَ وَإِخْوَتِهِۦٓ ءَايَـٰتٌ لِّلسَّآئِلِينَ٧إِذْ قَالُوا۟ لَيُوسُفُ وَأَخُوهُ أَحَبُّ إِلَىٰٓ أَبِينَا مِنَّا وَنَحْنُ عُصْبَةٌ إِنَّ أَبَانَا لَفِى ضَلَـٰلٍ مُّبِينٍ٨ٱقْتُلُوا۟ يُوسُفَ أَوِ ٱطْرَحُوهُ أَرْضًا يَخْلُ لَكُمْ وَجْهُ أَبِيكُمْ وَتَكُونُوا۟ مِنۢ بَعْدِهِۦ قَوْمًا صَـٰلِحِينَ٩قَالَ قَآئِلٌ مِّنْهُمْ لَا تَقْتُلُوا۟ يُوسُفَ وَأَلْقُوهُ فِى غَيَـٰبَتِ ٱلْجُبِّ يَلْتَقِطْهُ بَعْضُ ٱلسَّيَّارَةِ إِن كُنتُمْ فَـٰعِلِينَ٠١قَالُوا۟ يَـٰٓأَبَانَا مَا لَكَ لَا تَأْمَ۫نَّا عَلَىٰ يُوسُفَ وَإِنَّا لَهُۥ لَنَـٰصِحُونَ١١أَرْسِلْهُ مَعَنَا غَدًا يَرْتَعْ وَيَلْعَبْ وَإِنَّا لَهُۥ لَحَـٰفِظُونَ٢١قَالَ إِنِّى لَيَحْزُنُنِىٓ أَن تَذْهَبُوا۟ بِهِۦ وَأَخَافُ أَن يَأْكُلَهُ ٱلذِّئْبُ وَأَنتُمْ عَنْهُ غَـٰفِلُونَ٣١قَالُوا۟ لَئِنْ أَكَلَهُ ٱلذِّئْبُ وَنَحْنُ عُصْبَةٌ إِنَّآ إِذًا لَّخَـٰسِرُونَ٤١فَلَمَّا ذَهَبُوا۟ بِهِۦ وَأَجْمَعُوٓا۟ أَن يَجْعَلُوهُ فِى غَيَـٰبَتِ ٱلْجُبِّ ۚ وَأَوْحَيْنَآ إِلَيْهِ لَتُنَبِّئَنَّهُم بِأَمْرِهِمْ هَـٰذَا وَهُمْ لَا يَشْعُرُونَ٥١وَجَآءُوٓ أَبَاهُمْ عِشَآءً يَبْكُونَ٦١قَالُوا۟ يَـٰٓأَبَانَآ إِنَّا ذَهَبْنَا نَسْتَبِقُ وَتَرَكْنَا يُوسُفَ عِندَ مَتَـٰعِنَا فَأَكَلَهُ ٱلذِّئْبُ ۖ وَمَآ أَنتَ بِمُؤْمِنٍ لَّنَا وَلَوْ كُنَّا صَـٰدِقِينَ٧١وَجَآءُو عَلَىٰ قَمِيصِهِۦ بِدَمٍ كَذِبٍ ۚ قَالَ بَلْ سَوَّلَتْ لَكُمْ أَنفُسُكُمْ أَمْرًا ۖ فَصَبْرٌ جَمِيلٌ ۖ وَٱللَّهُ ٱلْمُسْتَعَانُ عَلَىٰ مَا تَصِفُونَ٨١
ವಾಸ್ತವದಲ್ಲಿ ಯೂಸುಫ್ ಮತ್ತು ಅವರ ಸಹೋದರರ ಚರಿತ್ರೆಯಲ್ಲಿ ಈ ಪ್ರಶ್ನಿಸುವವರಿಗೆ ದೊಡ್ಡ ನಿದರ್ಶನಗಳಿವೆ. ಈ ಕಥೆಯ ಆರಂಭ ಹೀಗಿದೆ-ಅವರ ಸಹೋದರರು ಪರಸ್ಪರ ಹೀಗೆ ಹೇಳಿದರು: ಈ ಯೂಸುಫನೂ ಅವನ ಸಹೋದರನೂ8 ನಮ್ಮ ತಂದೆಗೆ ನಮ್ಮೆಲ್ಲರಿಗಿಂತ ಹೆಚ್ಚು ಪ್ರಿಯರಾಗಿದ್ದಾರೆ. ವಸ್ತುತಃ ನಾವು ಒಂದು ಸಂಪೂರ್ಣ ಕೂಟವಾಗಿದ್ದೇವೆ. ನಿಜ ಹೇಳುವುದಾದರೆ ನಮ್ಮ ಪಿತೃವರ್ಯರು ಸ್ಪಷ್ಟ ದುರ್ಮಾರ್ಗದಲ್ಲಿದ್ದಾರೆ.9 ನಿಮ್ಮ ತಂದೆಯವರ ಒಲವು ನಿಮ್ಮ ಕಡೆಗೇ ಆಗುವಂತೆ ಮಾಡಲಿಕ್ಕಾಗಿ ಯೂಸುಫನನ್ನು ಕೊಂದು ಹಾಕಿರಿ, ಇಲ್ಲವೆ ಎಲ್ಲಾದರೂ ಎಸೆದುಬಿಡಿರಿ. ಈ ಕಾರ್ಯವನ್ನು ಮಾಡಿಯಾದ ಬಳಿಕ ಸಜ್ಜನರಾಗಿರಬೇಕು.10 ಆಗ ಅವರಲ್ಲೊಬ್ಬನು ಹೇಳಿದನು- "ಯೂಸುಫನನ್ನು ಕೊಲ್ಲಬೇಡಿರಿ. ಹಾಗೇನಾದರೂ ಮಾಡಲೇಬೇಕೆಂದಾದರೆ ಯಾವುದಾದರೊಂದು ಆಳವಾದ ಬಾವಿಯಲ್ಲಿ ಹಾಕಿಬಿಡಿರಿ. ಯಾವುದಾದರೂ ಪ್ರವಾಸಿ ತಂಡವು ಅವನನ್ನು ಎತ್ತಿಕೊಂಡೊಯ್ಯಬಹುದು." ಈ ನಿರ್ಧಾರ ಪ್ರಕಾರ ಅವರು ಹೋಗಿ ತಮ್ಮ ತಂದೆಯವರೊಡನೆ, "ಅಪ್ಪಾ, ನೀವು ಯೂಸುಫನ ಬಗ್ಗೆ ನಮ್ಮ ಮೇಲೆ ಭರವಸೆಯನ್ನೇಕೆ ಇರಿಸುವುದಿಲ್ಲ? ವಸ್ತುತಃ ನಾವು ಅವನ ನೈಜ ಹಿತಾಕಾಂಕ್ಷಿಗಳಾಗಿರುತ್ತೇವೆ. ನಾಳೆ ಅವನನ್ನು ನಮ್ಮ ಸಂಗಡ ಕಳುಹಿಸಿರಿ. ಏನಾದರೂ ತಿಂದು, ಆಡಿ, ನಲಿದು ಆನಂದ ಪಡುವನು. ನಾವು ಖಂಡಿತವಾಗಿಯೂ ಅವನ ರಕ್ಷಣೆಗೆ ಇದ್ದೇವೆ"11 ಎಂದರು. ಆಗ ತಂದೆ ಹೇಳಿದರು, "ನೀವು ಅವನನ್ನು ಕರೆದೊಯ್ಯುವುದು ನನ್ನನ್ನು ದುಃಖಕ್ಕೀಡು ಮಾಡುತ್ತಿದೆ. ನೀವು ಅವನ ಬಗ್ಗೆ ಅಜಾಗರೂಕರಾಗಿದ್ದಾಗ ಅವನನ್ನು ತೋಳ ತಿಂದುಬಿಡಬಹುದೆಂಬ ಭಯ ನನ್ನನ್ನು ಕಾಡುತ್ತಿದೆ." ಆಗ ಅವರು, "ನಾವು ಒಂದು ತಂಡವೇ ಆಗಿರುವಾಗ, ನಾವಿರುತ್ತ ಅವನನ್ನು ತೋಳ ತಿಂದರೆ, ನಾವು ಯಾವುದಕ್ಕೂ ಬೇಡದವರಾಗುವೆವು" ಎಂದರು. ಹೀಗೆ ಅವರು ಒತ್ತಾಯಪಡಿಸಿ ಅವರನ್ನು ಒಯ್ದಾಗ ಮತ್ತು ಅವರು ಅವರನ್ನು (ಯೂಸುಫರನ್ನು) ಒಂದು ಆಳವಾದ ಬಾವಿಗೆ ತಳ್ಳಿಬಿಡಬೇಕೆಂದು ಒಮ್ಮತದಿಂದ ನಿರ್ಧರಿಸಿದಾಗ, "ನೀನು ಇವರಿಗೆ ಇವರ ಈ ವರ್ತನೆಯ ಪರಿಣಾಮವನ್ನು ತೋರಿಸಿ ಕೊಡುವ ಸಂದರ್ಭವೊಂದು ಬರಲಿಕ್ಕಿದೆ. ಇವರು ಅದರ ಬಗ್ಗೆ ಪ್ರಜ್ಞಾಹೀನರಾಗಿದ್ದಾರೆ"12 ಎಂದು ನಾವು ಯೂಸುಫರಿಗೆ 'ದಿವ್ಯವಾಣಿ' ಮಾಡಿದೆವು. ಸಂಜೆ ಹೊತ್ತಿಗೆ ಅವರು ಅಳುತ್ತ, ತಮ್ಮ ತಂದೆಯ ಬಳಿಗೆ ಬಂದರು. "ಅಪ್ಪಾ, ನಾವು ಓಟದ ಸ್ಪರ್ಧೆಯಲ್ಲಿ ನಿರತರಾಗಿದ್ದೆವು. ಯೂಸುಫನನ್ನು ನಮ್ಮ ಸಾಮಾನುಗಳ ಹತ್ತಿರ ಬಿಟ್ಟಿದ್ದೆವು. ಅಷ್ಟರಲ್ಲಿ ತೋಳ ಬಂದು ಅವನನ್ನು ತಿಂದು ಬಿಟ್ಟಿತು. ನಾವು ಸತ್ಯವಂತರಾಗಿದ್ದರೂ ತಾವು ನಮ್ಮ ಮಾತನ್ನು ನಂಬಲಾರಿರಿ" ಎಂದರು. ಅವರು ಯೂಸುಫರ ಅಂಗಿಗೆ ಕೃತಕ ರಕ್ತ ಹಚ್ಚಿ ತಂದಿದ್ದರು. ಇದನ್ನು ಕೇಳಿ ಅವರ ತಂದೆ ಹೇಳಿದರು- "ಆದರೆ ನಿಮ್ಮ ಭ್ರಷ್ಟ ಚಿತ್ತಗಳು ನಿಮಗೆ ಒಂದು ಮಹಾ ಕಾರ್ಯವನ್ನು ಸುಲಭಗೊಳಿಸಿದುವು. ಚಿಂತಿಲ್ಲ, ನಾನು ಸಹಿಸಿಕೊಳ್ಳುವೆನು- ಚೆನ್ನಾಗಿ ಸಹಿಸಿಕೊಳ್ಳುವೆನು.13 ನಿಮ್ಮ ಕೃತಕ ಮಾತುಗಳ ಬಗ್ಗೆ ಅಲ್ಲಾಹನೊಡನೆ ಮಾತ್ರ ಸಹಾಯ ಯಾಚಿಸಬಹುದಷ್ಟೆ."14
12:19 - 20
وَجَآءَتْ سَيَّارَةٌ فَأَرْسَلُوا۟ وَارِدَهُمْ فَأَدْلَىٰ دَلْوَهُۥ ۖ قَالَ يَـٰبُشْرَىٰ هَـٰذَا غُلَـٰمٌ ۚ وَأَسَرُّوهُ بِضَـٰعَةً ۚ وَٱللَّهُ عَلِيمٌۢ بِمَا يَعْمَلُونَ٩١وَشَرَوْهُ بِثَمَنٍۭ بَخْسٍ دَرَٰهِمَ مَعْدُودَةٍ وَكَانُوا۟ فِيهِ مِنَ ٱلزَّٰهِدِينَ٠٢
ಅತ್ತ ಪ್ರವಾಸಿಗರ ತಂಡವೊಂದು ಬಂದಿತು. ಅದು ತನ್ನ ನೀರು ಕುಡಿಸುವವನನ್ನು ನೀರು ತರಲು ಕಳುಹಿಸಿತು. ಅವನು ತೊಟ್ಟಿಯನ್ನು ಬಾವಿಗೆ ಇಳಿಸಿದಾಗ (ಯೂಸುಫರನ್ನು ಕಂಡು) "ಶುಭವಾಗಲಿ! ಇಲ್ಲೊಬ್ಬ ಬಾಲಕನಿದ್ದಾನೆ" ಎಂದು ಕೂಗಿದನು. ಅವರು (ಪ್ರವಾಸೀ ತಂಡದವರು) ಅವರನ್ನು (ಯೂಸುಫರನ್ನು) ವ್ಯಾಪಾರದ ಸರಕಾಗಿ ಬಚ್ಚಿಟ್ಟುಕೊಂಡರು. ವಸ್ತುತಃ ಅವರು ಮಾಡುತ್ತಿದ್ದುದನ್ನು ಅಲ್ಲಾಹ್ ಬಲ್ಲವನಾಗಿದ್ದನು. ಕೊನೆಗೆ ಅವರು ಕೆಲವೇ ದಿರ್‌ಹಮ್‌ಗಳ ತುಚ್ಛ ಬೆಲೆಗೆ ಅವರನ್ನು ಮಾರಿದರು15 ಮತ್ತು ಅವರಿಗೆ ಅವರ ಬೆಲೆಯ ಬಗ್ಗೆ ಹೆಚ್ಚಿನ ನಿರೀಕ್ಷೆಯಿರಲಿಲ್ಲ.
12:21 - 22
وَقَالَ ٱلَّذِى ٱشْتَرَىٰهُ مِن مِّصْرَ لِٱمْرَأَتِهِۦٓ أَكْرِمِى مَثْوَىٰهُ عَسَىٰٓ أَن يَنفَعَنَآ أَوْ نَتَّخِذَهُۥ وَلَدًا ۚ وَكَذَٰلِكَ مَكَّنَّا لِيُوسُفَ فِى ٱلْأَرْضِ وَلِنُعَلِّمَهُۥ مِن تَأْوِيلِ ٱلْأَحَادِيثِ ۚ وَٱللَّهُ غَالِبٌ عَلَىٰٓ أَمْرِهِۦ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ١٢وَلَمَّا بَلَغَ أَشُدَّهُۥٓ ءَاتَيْنَـٰهُ حُكْمًا وَعِلْمًا ۚ وَكَذَٰلِكَ نَجْزِى ٱلْمُحْسِنِينَ٢٢
ಅವರನ್ನು ಖರೀದಿಸಿದ ಈಜಿಪ್ಟಿನವನು16 ತನ್ನ ಪತ್ನಿಯೊಡನೆ,17 "ಇವನನ್ನು ಚೆನ್ನಾಗಿ ನೋಡಿಕೋ, ಇವನು ನಮಗೆ ಫಲಕಾರಿಯಾಗಲೂ ಬಹುದು ಅಥವಾ ನಾವು ಇವನನ್ನು ದತ್ತುಪುತ್ರನನ್ನಾಗಿಯೂ ಮಾಡಿಕೊಳ್ಳಬಹುದು"!18 ಎಂದನು. ಈ ರೀತಿ ನಾವು ಯೂಸುಫರಿಗೆ ಆ ಭೂಭಾಗದಲ್ಲಿ ನೆಲೆಯೂರಲು ಅವಕಾಶ ಮಾಡಿ ಕೊಟ್ಟೆವು ಮತ್ತು ಅವರಿಗೆ ವಿಷಯಗಳನ್ನು ಗ್ರಹಿಸುವ ಜ್ಞಾನವನ್ನು ಕೊಡುವ ವ್ಯವಸ್ಥೆ ಮಾಡಿದೆವು.19 ಅಲ್ಲಾಹ್ ತನ್ನ ಕಾರ್ಯವನ್ನು ಮಾಡಲು ಸುಶಕ್ತನು. ಆದರೆ ಹೆಚ್ಚಿನವರು ತಿಳಿಯುವುದಿಲ್ಲ. ಅವರು ತಮ್ಮ ತುಂಬು ಯೌವನಕ್ಕೆ ತಲುಪಿದಾಗ ನಾವು ಅವರಿಗೆ ತೀರ್ಮಾನದ ಶಕ್ತಿ ಮತ್ತು ಜ್ಞಾನ ನೀಡಿದೆವು.20 ಸಜ್ಜನರಿಗೆ ನಾವು ಈ ರೀತಿ ಪ್ರತಿಫಲ ನೀಡುತ್ತೇವೆ.

وَرَٰوَدَتْهُ ٱلَّتِى هُوَ فِى بَيْتِهَا عَن نَّفْسِهِۦ وَغَلَّقَتِ ٱلْأَبْوَٰبَ وَقَالَتْ هَيْتَ لَكَ ۚ قَالَ مَعَاذَ ٱللَّهِ ۖ إِنَّهُۥ رَبِّىٓ أَحْسَنَ مَثْوَاىَ ۖ إِنَّهُۥ لَا يُفْلِحُ ٱلظَّـٰلِمُونَ٣٢وَلَقَدْ هَمَّتْ بِهِۦ ۖ وَهَمَّ بِهَا لَوْلَآ أَن رَّءَا بُرْهَـٰنَ رَبِّهِۦ ۚ كَذَٰلِكَ لِنَصْرِفَ عَنْهُ ٱلسُّوٓءَ وَٱلْفَحْشَآءَ ۚ إِنَّهُۥ مِنْ عِبَادِنَا ٱلْمُخْلَصِينَ٤٢وَٱسْتَبَقَا ٱلْبَابَ وَقَدَّتْ قَمِيصَهُۥ مِن دُبُرٍ وَأَلْفَيَا سَيِّدَهَا لَدَا ٱلْبَابِ ۚ قَالَتْ مَا جَزَآءُ مَنْ أَرَادَ بِأَهْلِكَ سُوٓءًا إِلَّآ أَن يُسْجَنَ أَوْ عَذَابٌ أَلِيمٌ٥٢قَالَ هِىَ رَٰوَدَتْنِى عَن نَّفْسِى ۚ وَشَهِدَ شَاهِدٌ مِّنْ أَهْلِهَآ إِن كَانَ قَمِيصُهُۥ قُدَّ مِن قُبُلٍ فَصَدَقَتْ وَهُوَ مِنَ ٱلْكَـٰذِبِينَ٦٢وَإِن كَانَ قَمِيصُهُۥ قُدَّ مِن دُبُرٍ فَكَذَبَتْ وَهُوَ مِنَ ٱلصَّـٰدِقِينَ٧٢فَلَمَّا رَءَا قَمِيصَهُۥ قُدَّ مِن دُبُرٍ قَالَ إِنَّهُۥ مِن كَيْدِكُنَّ ۖ إِنَّ كَيْدَكُنَّ عَظِيمٌ٨٢يُوسُفُ أَعْرِضْ عَنْ هَـٰذَا ۚ وَٱسْتَغْفِرِى لِذَنۢبِكِ ۖ إِنَّكِ كُنتِ مِنَ ٱلْخَاطِـِٔينَ٩٢
ಯೂಸುಫರು ವಾಸಿಸುತ್ತಿದ್ದ ಮನೆಯೊಡತಿ ಅವರ ಮೇಲೆ ಪ್ರೇಮ ಜಾಲ ಬೀಸತೊಡಗಿದಳು. ಒಂದು ದಿನ ಕದ ಮುಚ್ಚಿ, 'ಇತ್ತ ಬಾ' ಎಂದಳು. ಆಗ ಯೂಸುಫರು "ಅಲ್ಲಾಹನು ರಕ್ಷಿಸಲಿ. ನನ್ನ ಪ್ರಭು ನನಗೆ ಉತ್ತಮ ಸ್ಥಾನವನ್ನು ದಯಪಾಲಿಸಿದ್ದಾನೆ. (ಹೀಗಿರುವಾಗ ನಾನು ಈ ನೀಚ ಕಾರ್ಯವೆಸಗುವುದೇ?) ಇಂತಹ ಅಕ್ರಮಿಗಳು ಎಂದೂ ಯಶಸ್ವಿಗಳಾಗುವುದಿಲ್ಲ"21 ಎಂದರು. ಅವಳು ಅವರ ಕಡೆಗೆ ಮುಂದುವರಿದಳು. ಯೂಸುಫರೂ ತಮ್ಮ ಪ್ರಭುವಿನ ನಿದರ್ಶನವನ್ನು ಕಾಣದಿರುತ್ತಿದ್ದರೆ22 ಅವಳ ಕಡೆಗೆ ಮುಂದಾಗುತ್ತಿದ್ದರು. ಹೀಗಾದುದು ನಾವು ಅವರಿಂದ ಪಾಪ ಮತ್ತು ನಿರ್ಲಜ್ಜತನವನ್ನು ತೊಲಗಿಸಲಿಕ್ಕಾಗಿ,23 ವಾಸ್ತವದಲ್ಲಿ ಅವರು ನಮ್ಮ ಆಯ್ದ ದಾಸರಲ್ಲಿ ಸೇರಿದ್ದರು. ಕೊನೆಗೆ ಯೂಸುಫರು ಮುಂದಿನಿಂದಲೂ ಅವಳು ಹಿಂದಿನಿಂದಲೂ ಬಾಗಿಲಿನತ್ತ ಓಡಿದರು ಮತ್ತು ಹಿಂದಿನಿಂದ ಯೂಸುಫರ ಅಂಗಿಯನ್ನು (ಹಿಡಿದೆಳೆದು) ಹರಿದಳು. ಬಾಗಿಲಲ್ಲಿ ಅವರಿಬ್ಬರೂ ಅವಳ ಪತಿಯಿದ್ದುದನ್ನು ಕಂಡರು. ಅವನನ್ನು ಕಂಡೊಡನೆ ಆಕೆ ಹೇಳಿದಳು, "ನಿಮ್ಮ ಮನೆಯಾಕೆಯೊಂದಿಗೆ ನೀಚಕಾರ್ಯವೆಸಗಲಿಚ್ಛಿಸಿದವನ ಶಿಕ್ಷೆಯೇನು? ಇವನಿಗೆ ಸೆರೆಮನೆವಾಸ ಅಥವಾ ವೇದನಾತ್ಮಕ ದಂಡನೆಯ ವಿನಾ ಇನ್ನಾವ ಶಿಕ್ಷೆ ಇರಬಹುದು?" (ಆಗ) ಯೂಸುಫರು, "ಇವಳೇ ನನ್ನನ್ನು ಬಲೆಗೆ ಸಿಕ್ಕಿಸಲು ಯತ್ನಿಸಿದ್ದಳು" ಎಂದರು. ಆ ಸ್ತ್ರೀಯ ಕುಟುಂಬದವರಲ್ಲೊಬ್ಬನು (ಸಂದರ್ಭೋಚಿತ) ಸಾಕ್ಷ್ಯವನ್ನು ಹೀಗೆ ನೀಡಿದನು-24 "ಯೂಸುಫನ ಅಂಗಿ ಮುಂಭಾಗದಿಂದ ಹರಿದಿದ್ದರೆ ಸ್ತ್ರೀಯು ಸತ್ಯವಂತಳು ಮತ್ತು ಇವನು ಸುಳ್ಳುಗಾರ. ಇವನ ಅಂಗಿ ಹಿಂಭಾಗದಿಂದ ಹರಿದಿದ್ದರೆ ಸ್ತ್ರೀಯು ಸುಳ್ಳುಗಾರ್ತಿ ಮತ್ತು ಇವನು ಸತ್ಯವಂತ."25 ಆಕೆಯ ಪತಿಯು ಯೂಸುಫರ ಅಂಗಿಯು ಹಿಂಭಾಗದಿಂದ ಹರಿದಿರುವುದನ್ನು ಕಂಡಾಗ ಹೀಗೆಂದನು, "ನಿಮ್ಮ (ಸ್ತ್ರೀಯರ) ಕುತಂತ್ರಗಳಿವು. ನಿಜಕ್ಕೂ ನಿಮ್ಮ ಕುತಂತ್ರಗಳು ಕಂಟಕ ಪ್ರಾಯವಾಗಿವೆ. ಯೂಸುಫ್! ನೀನು ಈ ವಿಷಯವನ್ನು ಕಡೆಗಣಿಸು ಮತ್ತು ಓ ಸ್ತ್ರೀಯೇ! ನೀನು ನಿನ್ನ ಅಪರಾಧಕ್ಕಾಗಿ ಕ್ಷಮೆ ಯಾಚಿಸು. ವಾಸ್ತವದಲ್ಲಿ ನೀನೇ ಅಪರಾಧಿಯಾಗಿದ್ದಿ."25(ಅ)
12:30 - 34
۞ وَقَالَ نِسْوَةٌ فِى ٱلْمَدِينَةِ ٱمْرَأَتُ ٱلْعَزِيزِ تُرَٰوِدُ فَتَىٰهَا عَن نَّفْسِهِۦ ۖ قَدْ شَغَفَهَا حُبًّا ۖ إِنَّا لَنَرَىٰهَا فِى ضَلَـٰلٍ مُّبِينٍ٠٣فَلَمَّا سَمِعَتْ بِمَكْرِهِنَّ أَرْسَلَتْ إِلَيْهِنَّ وَأَعْتَدَتْ لَهُنَّ مُتَّكَـًٔا وَءَاتَتْ كُلَّ وَٰحِدَةٍ مِّنْهُنَّ سِكِّينًا وَقَالَتِ ٱخْرُجْ عَلَيْهِنَّ ۖ فَلَمَّا رَأَيْنَهُۥٓ أَكْبَرْنَهُۥ وَقَطَّعْنَ أَيْدِيَهُنَّ وَقُلْنَ حَـٰشَ لِلَّهِ مَا هَـٰذَا بَشَرًا إِنْ هَـٰذَآ إِلَّا مَلَكٌ كَرِيمٌ١٣قَالَتْ فَذَٰلِكُنَّ ٱلَّذِى لُمْتُنَّنِى فِيهِ ۖ وَلَقَدْ رَٰوَدتُّهُۥ عَن نَّفْسِهِۦ فَٱسْتَعْصَمَ ۖ وَلَئِن لَّمْ يَفْعَلْ مَآ ءَامُرُهُۥ لَيُسْجَنَنَّ وَلَيَكُونًا مِّنَ ٱلصَّـٰغِرِينَ٢٣قَالَ رَبِّ ٱلسِّجْنُ أَحَبُّ إِلَىَّ مِمَّا يَدْعُونَنِىٓ إِلَيْهِ ۖ وَإِلَّا تَصْرِفْ عَنِّى كَيْدَهُنَّ أَصْبُ إِلَيْهِنَّ وَأَكُن مِّنَ ٱلْجَـٰهِلِينَ٣٣فَٱسْتَجَابَ لَهُۥ رَبُّهُۥ فَصَرَفَ عَنْهُ كَيْدَهُنَّ ۚ إِنَّهُۥ هُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ٤٣
ನಗರದ ವನಿತೆಯರು ಪರಸ್ಪರ ಚರ್ಚಿಸತೊಡಗಿದರು- "ಅಝೀಝನ ಪತ್ನಿ ತನ್ನ ತರುಣ ಗುಲಾಮನನ್ನು ಮೋಹಪಾಶದಲ್ಲಿ ಬಿಗಿಯಲು ಯತ್ನಿಸುತ್ತಿದ್ದಾಳೆ. ಪ್ರೇಮವು ಅವಳನ್ನು ಅನಿಯಂತ್ರಿತಗೊಳಿಸಿದೆ. ನಮ್ಮ ದೃಷ್ಟಿಯಲ್ಲಿ ಅವಳು ಸುವ್ಯಕ್ತ ತಪ್ಪು ಹಾದಿಯಲ್ಲಿದ್ದಾಳೆ." ಅವಳು ಅವರ ಈ ಕುಟಿಲ ಮಾತುಗಳನ್ನು ಕೇಳಿದಾಗ ಅವರನ್ನು ಬರ ಹೇಳಿದಳು ಮತ್ತು ಅವರಿಗಾಗಿ ಸುಪ್ಪತ್ತಿಗೆಗಳಿಂದ ಸಜ್ಜಾದ ಸಭೆಯೊಂದನ್ನು ಏರ್ಪಡಿಸಿದಳು.26 ಔತಣದಲ್ಲಿ ಪ್ರತಿಯೊಬ್ಬಳ ಮುಂದೆ ಒಂದೊಂದು ಚೂರಿಯನ್ನಿರಿಸಿದಳು. (ಆ ಸ್ತ್ರೀಯರು ಹಣ್ಣುಗಳನ್ನು ಕೊಯ್ದು ತಿನ್ನುತ್ತಲಿದ್ದ ಅದೇ ವೇಳೆಯಲ್ಲಿ) ಅವಳು ಯೂಸುಫರಿಗೆ ಅವರ ಮುಂದೆ ನಡೆದು ಬರುವಂತೆ ಸೂಚನೆ ಕೊಟ್ಟಳು. ಆ ಸ್ತ್ರೀಯರ ದೃಷ್ಟಿ ಅವರ ಮೇಲೆ ಬಿದ್ದಾಗ ಅವರು ದಂಗಾದರು ಮತ್ತು ತಮ್ಮ ಕೈಗಳನ್ನೇ ಕೊಯ್ದುಕೊಂಡರು. "ಅಲ್ಲ, ಈತನು ಎಷ್ಟು ಮಾತ್ರಕ್ಕೂ ಮಾನವನಲ್ಲ. ಇದಾರೋ ಸನ್ಮಾನ್ಯ ದೇವಚರ" ಎಂದು ಭಾವಪರವಶರಾಗಿ ಕೂಗಿಕೊಂಡರು. ಅಝೀಝನ ಪತ್ನಿ ಹೇಳಿದಳು, "ಕಂಡಿರಾ! ಯಾರ ವಿಷಯವಾಗಿ ನೀವು ನನ್ನ ಕುರಿತು ಆಕ್ಷೇಪದ ಮಾತುಗಳನ್ನಾಡುತ್ತಿದ್ದಿರೋ ಆ ವ್ಯಕ್ತಿ ಇವನೇ. ನಿಸ್ಸಂದೇಹವಾಗಿಯೂ ನಾನು ಇವನನ್ನು ವಶೀಕರಿಸಲು ಪ್ರಯತ್ನಿಸಿದ್ದೆ. ಆದರೆ ಇವನು ತಪ್ಪಿಸಿಕೊಂಡನು. ಇವನು ನನ್ನ ಮಾತು ಕೇಳದಿದ್ದರೆ, ಸೆರೆಮನೆ ಸೇರುವನು ಮತ್ತು ಅಪಮಾನಕ್ಕೊಳಗಾಗುವನು."27 ಯೂಸುಫ್ ಹೇಳಿದರು, "ಓ ನನ್ನ ಪ್ರಭೂ, ಇವರು ನನ್ನಿಂದ ಬಯಸುವ ಕಾರ್ಯಕ್ಕಿಂತ ನನಗೆ ಸೆರೆಮನೆಯೇ ಲೇಸು. ನೀನು ಇವರ ಕುತಂತ್ರಗಳನ್ನು ನನ್ನಿಂದ ನೀಗಿಸದಿದ್ದರೆ ನಾನು ಅವರ ಬಲೆಯಲ್ಲಿ ಬಿದ್ದು ಹೋಗುವೆನು ಮತ್ತು ಅಜ್ಞಾನಿಗಳಲ್ಲಿ ಸೇರಿ ಹೋಗುವೆನು."28 ಅವರ ಪ್ರಭು ಅವರ ಪ್ರಾರ್ಥನೆಯನ್ನು ಸ್ವೀಕರಿಸಿದನು ಮತ್ತು ಆ ಸ್ತ್ರೀಯರ ಕುತಂತ್ರಗಳನ್ನು ಅವರಿಂದ ನೀಗಿಸಿದನು.29 ನಿಶ್ಚಯವಾಗಿಯೂ ಅವನು ಸರ್ವಶ್ರುತ ಮತ್ತು ಸರ್ವಜ್ಞನಾಗಿದ್ದಾನೆ.

ثُمَّ بَدَا لَهُم مِّنۢ بَعْدِ مَا رَأَوُا۟ ٱلْـَٔايَـٰتِ لَيَسْجُنُنَّهُۥ حَتَّىٰ حِينٍ٥٣
ತರುವಾಯ ಒಂದು ಕಾಲಾವಧಿಯವರೆಗೆ ಇವರನ್ನು ಸೆರೆಯಲ್ಲಿಡಬೇಕೆಂದು ಅವರಿಗೆ ತೋರಿತು. ವಸ್ತುತಃ ಅವರು (ಇವರ ಪಾವಿತ್ರ್ಯ ಮತ್ತು ಸ್ವತಃ ತಮ್ಮ ಸ್ತ್ರೀಯರ ದುರ್ಗುಣಗಳ) ಸುಸ್ಪಷ್ಟ ದೃಷ್ಟಾಂತಗಳನ್ನು ಕಂಡಿದ್ದರು.30
12:36 - 41
وَدَخَلَ مَعَهُ ٱلسِّجْنَ فَتَيَانِ ۖ قَالَ أَحَدُهُمَآ إِنِّىٓ أَرَىٰنِىٓ أَعْصِرُ خَمْرًا ۖ وَقَالَ ٱلْـَٔاخَرُ إِنِّىٓ أَرَىٰنِىٓ أَحْمِلُ فَوْقَ رَأْسِى خُبْزًا تَأْكُلُ ٱلطَّيْرُ مِنْهُ ۖ نَبِّئْنَا بِتَأْوِيلِهِۦٓ ۖ إِنَّا نَرَىٰكَ مِنَ ٱلْمُحْسِنِينَ٦٣قَالَ لَا يَأْتِيكُمَا طَعَامٌ تُرْزَقَانِهِۦٓ إِلَّا نَبَّأْتُكُمَا بِتَأْوِيلِهِۦ قَبْلَ أَن يَأْتِيَكُمَا ۚ ذَٰلِكُمَا مِمَّا عَلَّمَنِى رَبِّىٓ ۚ إِنِّى تَرَكْتُ مِلَّةَ قَوْمٍ لَّا يُؤْمِنُونَ بِٱللَّهِ وَهُم بِٱلْـَٔاخِرَةِ هُمْ كَـٰفِرُونَ٧٣وَٱتَّبَعْتُ مِلَّةَ ءَابَآءِىٓ إِبْرَٰهِيمَ وَإِسْحَـٰقَ وَيَعْقُوبَ ۚ مَا كَانَ لَنَآ أَن نُّشْرِكَ بِٱللَّهِ مِن شَىْءٍ ۚ ذَٰلِكَ مِن فَضْلِ ٱللَّهِ عَلَيْنَا وَعَلَى ٱلنَّاسِ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَشْكُرُونَ٨٣يَـٰصَـٰحِبَىِ ٱلسِّجْنِ ءَأَرْبَابٌ مُّتَفَرِّقُونَ خَيْرٌ أَمِ ٱللَّهُ ٱلْوَٰحِدُ ٱلْقَهَّارُ٩٣مَا تَعْبُدُونَ مِن دُونِهِۦٓ إِلَّآ أَسْمَآءً سَمَّيْتُمُوهَآ أَنتُمْ وَءَابَآؤُكُم مَّآ أَنزَلَ ٱللَّهُ بِهَا مِن سُلْطَـٰنٍ ۚ إِنِ ٱلْحُكْمُ إِلَّا لِلَّهِ ۚ أَمَرَ أَلَّا تَعْبُدُوٓا۟ إِلَّآ إِيَّاهُ ۚ ذَٰلِكَ ٱلدِّينُ ٱلْقَيِّمُ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ٠٤يَـٰصَـٰحِبَىِ ٱلسِّجْنِ أَمَّآ أَحَدُكُمَا فَيَسْقِى رَبَّهُۥ خَمْرًا ۖ وَأَمَّا ٱلْـَٔاخَرُ فَيُصْلَبُ فَتَأْكُلُ ٱلطَّيْرُ مِن رَّأْسِهِۦ ۚ قُضِىَ ٱلْأَمْرُ ٱلَّذِى فِيهِ تَسْتَفْتِيَانِ١٤
ಸೆರೆಮನೆಯೊಳಗೆ31 ಇನ್ನಿಬ್ಬರು ಗುಲಾಮರೂ ಇವರೊಂದಿಗೆ ಪ್ರವೇಶಿಸಿದರು.32 ಒಂದು ದಿನ ಅವರಲ್ಲೊಬ್ಬನು "ನಾನು ಸಾರಾಯಿ ಭಟ್ಟಿ ಇಳಿಸುತ್ತಿರುವುದನ್ನು ಸ್ವಪ್ನದಲ್ಲಿ ಕಂಡೆನು" ಎಂದನು. ಇನ್ನೊಬ್ಬನು, "ನನ್ನ ತಲೆಯ ಮೇಲೆ ರೊಟ್ಟಿಯಿರಿಸಿರುವುದನ್ನೂ ಪಕ್ಷಿಗಳು ಅವುಗಳನ್ನು ತಿನ್ನುತ್ತಿರುವುದನ್ನೂ ನಾನು ಕಂಡೆನು" ಎಂದನು. ನಮಗೆ ಇದರ ಮರ್ಮ ತಿಳಿಸಬೇಕೆಂದೂ ನೀವೊಬ್ಬ ಸಜ್ಜನರಾಗಿರುವುದನ್ನು ನಾವು ಕಾಣುತ್ತೇವೆಂದೂ ಇಬ್ಬರೂ ಹೇಳಿದರು.33 ಯೂಸುಫ್ (ಹೀಗೆ) ಹೇಳಿದರು- "ಇಲ್ಲಿ ನಿಮಗೆ ಸಿಗುತ್ತಿರುವ ಊಟ ಬರುವುದಕ್ಕೆ ಮುಂಚೆ ನಾನು ನಿಮಗೆ ಈ ಸ್ವಪ್ನಗಳ ಫಲವನ್ನು ಹೇಳುವೆನು. ಇದು ನನ್ನ ಪ್ರಭು ನನಗೆ ದಯಪಾಲಿಸಿರುವ ಜ್ಞಾನಗಳಲ್ಲೊಂದು. ವಾಸ್ತವದಲ್ಲಿ ನಾನು ಅಲ್ಲಾಹನ ಮೇಲೆ ವಿಶ್ವಾಸವಿಡದವರ ಮತ್ತು ಪರಲೋಕವನ್ನು ನಿರಾಕರಿಸುವವರ ಮಾರ್ಗವನ್ನು ಬಿಟ್ಟು ನಮ್ಮ ಪೂರ್ವಜರಾಗಿರುವ ಇಬ್ರಾಹೀಮ್, ಇಸ್‌ಹಾಕ್ ಮತ್ತು ಯಾಕೂಬರ ಮಾರ್ಗವನ್ನು ಅವಲಂಬಿಸಿದ್ದೇನೆ. ಅಲ್ಲಾಹನೊಂದಿಗೆ ಇತರರನ್ನು ಸಹಭಾಗಿಗಳನ್ನಾಗಿಸುವುದು ನಮ್ಮ ಕೆಲಸವಲ್ಲ. ವಾಸ್ತವದಲ್ಲಿ ಇದು (ಅಲ್ಲಾಹನು ನಮ್ಮನ್ನು ತನ್ನ ಹೊರತು ಇನ್ನಾರದೇ ದಾಸರಾಗಿ ಮಾಡದಿರುವುದು) ನಮ್ಮ ಮೇಲೆ ಮತ್ತು ಸಕಲ ಮಾನವರ ಮೇಲೆ ಅಲ್ಲಾಹನ ಅನುಗ್ರಹವಾಗಿದೆ. ಆದರೆ ಹೆಚ್ಚಿನವರು ಕೃತಜ್ಞತೆ ಸಲ್ಲಿಸುವುದಿಲ್ಲ. ಓ ಸೆರೆಮನೆಯ ನನ್ನ ಸಂಗಾತಿಗಳೇ, "ಅನೇಕ ಭಿನ್ನ ಭಿನ್ನ ಪ್ರಭುಗಳು ಲೇಸೋ ಅಥವಾ ಸರ್ವ ಸಮರ್ಥನಾದ ಏಕ ಅಲ್ಲಾಹನೋ ಎಂದು ನೀವು ಸ್ವತಃ ಯೋಚಿಸಿರಿ. ಅವನನ್ನು ಬಿಟ್ಟು ನೀವು ಯಾರ ದಾಸ್ಯ-ಆರಾಧನೆ ಮಾಡುತ್ತಿರುವಿರೋ ಅವರೆಲ್ಲಾ ನೀವೂ ನಿಮ್ಮ ಪೂರ್ವಜರೂ ಇರಿಸಿಕೊಂಡಿರುವ ಕೆಲವು ಹೆಸರುಗಳ ವಿನಾ ಬೇರೇನೂ ಅಲ್ಲ. ಅಲ್ಲಾಹನು ಅವುಗಳ ಬಗ್ಗೆ ಯಾವ ಪ್ರಮಾಣವನ್ನೂ ಇಳಿಸಿರುವುದಿಲ್ಲ. ಆಜ್ಞಾಧಿಕಾರವು ಅಲ್ಲಾಹನಿಗೆ ಹೊರತು ಇನ್ನಾರಿಗೂ ಇಲ್ಲ. ಅವನ ಹೊರತು ನೀವು ಯಾರದೇ ದಾಸ್ಯ-ಆರಾಧನೆ ಮಾಡಬಾರದೆಂದು ಅವನ ಕಟ್ಟಪ್ಪಣೆ ಇದೆ. ಇದೇ ಅಪ್ಪಟ ಋಜು ಜೀವನ ಮಾರ್ಗ. ಆದರೆ ಹೆಚ್ಚಿನ ಜನರು ಅರಿಯುವುದಿಲ್ಲ. ಓ ಸೆರೆಮನೆಯ ನನ್ನ ಸಂಗಾತಿಗಳೇ, ನಿಮ್ಮ ಸ್ವಪ್ನ ಫಲವಿದು: ನಿಮ್ಮಲ್ಲೋರ್ವನು ತನ್ನ ಪ್ರಭುವಿಗೆ (ಈಜಿಪ್ಟಿನ ದೊರೆಗೆ) ಮದ್ಯ ಕುಡಿಸುವನು. ಇನ್ನೊಬ್ಬನು ಗಲ್ಲಿಗೇರಿಸಲ್ಪಡುವನು ಮತ್ತು ಪಕ್ಷಿಗಳು ಅವನ ತಲೆಯನ್ನು ಕುಕ್ಕಿ ಕುಕ್ಕಿ ತಿನ್ನುವುವು. ನೀವು ಕೇಳುತ್ತಲಿದ್ದ ವಿಷಯದ ತೀರ್ಮಾನ ಆಗಿ ಹೋಯಿತು."34
12:42 - 42
وَقَالَ لِلَّذِى ظَنَّ أَنَّهُۥ نَاجٍ مِّنْهُمَا ٱذْكُرْنِى عِندَ رَبِّكَ فَأَنسَىٰهُ ٱلشَّيْطَـٰنُ ذِكْرَ رَبِّهِۦ فَلَبِثَ فِى ٱلسِّجْنِ بِضْعَ سِنِينَ٢٤
ಅನಂತರ, ಅವರ ಪೈಕಿ ಯಾರು ಬಿಡುಗಡೆಯಾಗುವನೆಂಬ ನಿರೀಕ್ಷೆಯಿತ್ತೋ ಅವನೊಡನೆ ಯೂಸುಫರು, "ನಿನ್ನ ಪ್ರಭುವಿನೊಡನೆ (ಈಜಿಪ್ಟಿನ ದೊರೆ) ನನ್ನ ಪ್ರಸ್ತಾಪವೆತ್ತಬೇಕು" ಎಂದು ಹೇಳಿದರು. ಆದರೆ ಶೈತಾನನು ಅವನಿಗೆ ಮರವೆಯನ್ನುಂಟು ಮಾಡಿಬಿಟ್ಟನು. ಅವನು ತನ್ನ ಪ್ರಭುವಿನೊಡನೆ (ಈಜಿಪ್ಟಿನ ದೊರೆ) ಯೂಸುಫರ ಪ್ರಸ್ತಾಪವೆತ್ತಲು ಮರೆತೇ ಬಿಟ್ಟನು. ಹೀಗೆ ಅವರು ಅನೇಕ ವರ್ಷಗಳ ವರೆಗೆ ಸೆರೆಮನೆಯಲ್ಲೇ ಉಳಿದುಬಿಟ್ಟರು.35
12:43 - 44
وَقَالَ ٱلْمَلِكُ إِنِّىٓ أَرَىٰ سَبْعَ بَقَرَٰتٍ سِمَانٍ يَأْكُلُهُنَّ سَبْعٌ عِجَافٌ وَسَبْعَ سُنۢبُلَـٰتٍ خُضْرٍ وَأُخَرَ يَابِسَـٰتٍ ۖ يَـٰٓأَيُّهَا ٱلْمَلَأُ أَفْتُونِى فِى رُءْيَـٰىَ إِن كُنتُمْ لِلرُّءْيَا تَعْبُرُونَ٣٤قَالُوٓا۟ أَضْغَـٰثُ أَحْلَـٰمٍ ۖ وَمَا نَحْنُ بِتَأْوِيلِ ٱلْأَحْلَـٰمِ بِعَـٰلِمِينَ٤٤
ಒಂದು ದಿನ36 ಅರಸನು, "ನಾನು ಸ್ವಪ್ನದಲ್ಲಿ ಏಳು ಕೊಬ್ಬಿದ ಹಸುಗಳನ್ನು ಏಳು ಕೃಶವಾದ ಹಸುಗಳು ತಿನ್ನುತ್ತಿರುವುದನ್ನೂ ಧಾನ್ಯದ ಏಳು ತೆನೆಗಳು ಹಸುರಾಗಿದ್ದು ಇತರ ಏಳು ತೆನೆಗಳು ಒಣಗಿರುವುದನ್ನೂ ಕಂಡಿರುತ್ತೇನೆ. ಓ ನನ್ನ ಆಸ್ಥಾನದವರೇ, ನೀವು ಸ್ವಪ್ನಗಳ ಫಲ ಬಲ್ಲವರಾಗಿದ್ದರೆ, ನನಗೆ ಈ ಸ್ವಪ್ನದ ಫಲವನ್ನು ಹೇಳಿರಿ" ಎಂದನು.37 ಆಗ ಜನರು, "ಗೊಂದಲಪೂರ್ಣ ಸ್ವಪ್ನಗಳಿವು. ಇಂತಹ ಸ್ವಪ್ನಗಳ ಫಲ ನಮಗೆ ತಿಳಿಯದು" ಎಂದರು.
12:45 - 45
وَقَالَ ٱلَّذِى نَجَا مِنْهُمَا وَٱدَّكَرَ بَعْدَ أُمَّةٍ أَنَا۠ أُنَبِّئُكُم بِتَأْوِيلِهِۦ فَأَرْسِلُونِ٥٤
ಆ ಇಬ್ಬರು ಸೆರೆಯಾಳುಗಳ ಪೈಕಿ ಬದುಕಿ ಉಳಿದಿದ್ದವನಿಗೆ ದೀರ್ಘ ಕಾಲದ ತರುವಾಯ ಈಗ ಆ ಮಾತು ನೆನಪಿಗೆ ಬಂತು. ಅವನು, "ನಿಮಗೆ ಇದರ ಫಲ ನಾನು ಹೇಳುತ್ತೇನೆ. ನನ್ನನ್ನು ಸ್ವಲ್ಪ ಹೊತ್ತಿನ ಮಟ್ಟಿಗೆ (ಸೆರೆಮನೆಯಲ್ಲಿರುವ ಯೂಸುಫರ ಬಳಿಗೆ) ಕಳುಹಿಸಿರಿ"38 ಎಂದನು.
12:46 - 49
يُوسُفُ أَيُّهَا ٱلصِّدِّيقُ أَفْتِنَا فِى سَبْعِ بَقَرَٰتٍ سِمَانٍ يَأْكُلُهُنَّ سَبْعٌ عِجَافٌ وَسَبْعِ سُنۢبُلَـٰتٍ خُضْرٍ وَأُخَرَ يَابِسَـٰتٍ لَّعَلِّىٓ أَرْجِعُ إِلَى ٱلنَّاسِ لَعَلَّهُمْ يَعْلَمُونَ٦٤قَالَ تَزْرَعُونَ سَبْعَ سِنِينَ دَأَبًا فَمَا حَصَدتُّمْ فَذَرُوهُ فِى سُنۢبُلِهِۦٓ إِلَّا قَلِيلًا مِّمَّا تَأْكُلُونَ٧٤ثُمَّ يَأْتِى مِنۢ بَعْدِ ذَٰلِكَ سَبْعٌ شِدَادٌ يَأْكُلْنَ مَا قَدَّمْتُمْ لَهُنَّ إِلَّا قَلِيلًا مِّمَّا تُحْصِنُونَ٨٤ثُمَّ يَأْتِى مِنۢ بَعْدِ ذَٰلِكَ عَامٌ فِيهِ يُغَاثُ ٱلنَّاسُ وَفِيهِ يَعْصِرُونَ٩٤
ಅವನು ಹೋಗಿ ಯೂಸುಫರೊಡನೆ, "ಓ, ಸಾದ್ಯಂತ ಸತ್ಯಸಂಧರಾಗಿರುವವರೇ39, ಏಳು ಕೊಬ್ಬಿದ ಹಸುಗಳನ್ನು ಏಳು ಕೃಶವಾದ ಹಸುಗಳು ತಿನ್ನುತ್ತಿವೆ ಮತ್ತು ಏಳು ತೆನೆಗಳು ಹಸುರಾಗಿಯೂ ಇನ್ನು ಏಳು ತೆನೆಗಳು ಒಣಗಿಯೂ ಇವೆ. ಈ ಸ್ವಪ್ನದ ಫಲವನ್ನು ನನಗೆ ಹೇಳಿರಿ. ಹಾಗಾದರೆ ನಾನು ಜನರ ಕಡೆಗೆ ಮರಳಬಹುದು ಮತ್ತು ಅವರು ತಿಳಿದುಕೊಳ್ಳಬಹುದು"40 ಎಂದನು. ಯೂಸುಫ್ ಹೇಳಿದರು- "ಏಳು ವರ್ಷಗಳ ತನಕ ನೀವು ನಿರಂತರವಾಗಿ ಕೃಷಿ ಮಾಡುತ್ತಿರುವಿರಿ. ಆ ಕಾಲಾವಧಿಯಲ್ಲಿ ನೀವು ಕೊಯ್ಯುವ ಬೆಳೆಯಿಂದ ನಿಮ್ಮ ಆಹಾರಕ್ಕೆ ಬೇಕಾದಷ್ಟು ಅಂಶವನ್ನು ತೆಗೆದು ಉಳಿದುದನ್ನು ಅದರ ತೆನೆಗಳಲ್ಲೇ ಇರಗೊಡಿರಿ. ಅನಂತರ ಏಳು ಕಠಿಣ ವರ್ಷಗಳು ಬರುವುವು. ಆಗ ಆ ಕಾಲಕ್ಕಾಗಿ ಸಂಗ್ರಹಿಸಿಟ್ಟುದನ್ನು ಉಣ್ಣಬಹುದು. ಒಂದು ವೇಳೆ ಉಳಿಯುವುದಾದರೆ ನೀವು ಜೋಪಾನವಾಗಿರಿಸಿದುದು ಮಾತ್ರ. ಅನಂತರ ಬರುವ ಒಂದು ವರ್ಷದಲ್ಲಿ ಕೃಪಾವೃಷ್ಟಿಯ ಮೂಲಕ ಜನರ ಸಂಕಷ್ಟ ನಿವಾರಣೆಯಾಗುವುದು ಮತ್ತು ಅವರು ರಸ ಹಿಂಡುವರು."41
12:50 - 51
وَقَالَ ٱلْمَلِكُ ٱئْتُونِى بِهِۦ ۖ فَلَمَّا جَآءَهُ ٱلرَّسُولُ قَالَ ٱرْجِعْ إِلَىٰ رَبِّكَ فَسْـَٔلْهُ مَا بَالُ ٱلنِّسْوَةِ ٱلَّـٰتِى قَطَّعْنَ أَيْدِيَهُنَّ ۚ إِنَّ رَبِّى بِكَيْدِهِنَّ عَلِيمٌ٠٥قَالَ مَا خَطْبُكُنَّ إِذْ رَٰوَدتُّنَّ يُوسُفَ عَن نَّفْسِهِۦ ۚ قُلْنَ حَـٰشَ لِلَّهِ مَا عَلِمْنَا عَلَيْهِ مِن سُوٓءٍ ۚ قَالَتِ ٱمْرَأَتُ ٱلْعَزِيزِ ٱلْـَٔـٰنَ حَصْحَصَ ٱلْحَقُّ أَنَا۠ رَٰوَدتُّهُۥ عَن نَّفْسِهِۦ وَإِنَّهُۥ لَمِنَ ٱلصَّـٰدِقِينَ١٥
ಅರಸನು, "ಅವರನ್ನು ನನ್ನ ಬಳಿಗೆ ಕರೆದು ತನ್ನಿರಿ" ಎಂದನು. ಆದರೆ ರಾಜದೂತನು ಯೂಸುಫರ ಬಳಿಗೆ ಬಂದಾಗ ಅವರು, (ಹೀಗೆ) ಹೇಳಿದರು,42 "ನಿನ್ನ ಪ್ರಭುವಿನ ಬಳಿಗೆ ನೀನು ಹಿಂದಿರುಗು ಮತ್ತು ತಮ್ಮ ಕೈಗಳನ್ನು ಕೊಯ್ದುಕೊಂಡ ಸ್ತ್ರೀಯರ ಸಂಗತಿಯೇನೆಂದು ಅವನೊಡನೆ ಕೇಳು. ನನ್ನ ಪ್ರಭುವಂತು ಅವರ ಕುತಂತ್ರಗಳನ್ನು ತಿಳಿದೇ ಇದ್ದಾನೆ."43 ಆ ಪ್ರಕಾರ ಅರಸನು ಆ ಸ್ತ್ರೀಯರೊಡನೆ,44 "ನೀವು ಯೂಸುಫರನ್ನು ವಶೀಕರಿಸಲು ಪ್ರಯತ್ನಿಸಿದಾಗಿನ ನಿಮ್ಮ ಅನುಭವವೇನು?" ಎಂದು ಕೇಳಿದನು. ಎಲ್ಲರೂ ಒಕ್ಕೊರಳಿನಿಂದ, "ಎಷ್ಟಕ್ಕೂ ಇಲ್ಲ, ನಾವು ಅವರಲ್ಲಿ ಯಾವ ದೋಷವನ್ನೂ ಕಾಣಲಿಲ್ಲ" ಎಂದರು. ಅಝೀಝನ ಪತ್ನಿಯು, "ಈಗ ಸತ್ಯವು ಬಯಲಾಗಿದೆ. ಅವರನ್ನು ಪುಸಲಾಯಿಸಲು ಪ್ರಯತ್ನಿಸಿದವಳು ನಾನೇ ಆಗಿದ್ದೆ. ಖಂಡಿತವಾಗಿಯೂ ಅವರು ಸತ್ಯಸಂಧರು" ಎಂದು ಹೇಳಿದಳು.45
12:52 - 53
ذَٰلِكَ لِيَعْلَمَ أَنِّى لَمْ أَخُنْهُ بِٱلْغَيْبِ وَأَنَّ ٱللَّهَ لَا يَهْدِى كَيْدَ ٱلْخَآئِنِينَ٢٥۞ وَمَآ أُبَرِّئُ نَفْسِىٓ ۚ إِنَّ ٱلنَّفْسَ لَأَمَّارَةٌۢ بِٱلسُّوٓءِ إِلَّا مَا رَحِمَ رَبِّىٓ ۚ إِنَّ رَبِّى غَفُورٌ رَّحِيمٌ٣٥
ಯೂಸುಫ್ ಹೇಳಿದರು,46 "ಇದರ ಉದ್ದೇಶವು ನಾನು ತೆರೆಯ ಮರೆಯಲ್ಲಿ ಅಝೀಝನಿಗೆ ವಿಶ್ವಾಸಘಾತಕವೆಸಗಲಿಲ್ಲವೆಂದೂ ಅಲ್ಲಾಹನು ವಿಶ್ವಾಸಘಾತಕರ ಕುತಂತ್ರವನ್ನು ಸಫಲಗೊಳಿಸಲು ಬಿಡುವುದಿಲ್ಲವೆಂದೂ (ಅಝೀಝ್) ತಿಳಿದುಕೊಳ್ಳಬೇಕೆಂಬುದಾಗಿತ್ತು. ನನ್ನ ಚಿತ್ತವು ನಿರ್ದೋಷಿಯೆಂದು ನಾನೇನೂ ಹೇಳಿಕೊಳ್ಳುವುದಿಲ್ಲ. ನನ್ನ ಪ್ರಭುವಿನ ಕೃಪೆಯಿಲ್ಲದೆ ಹೋದರೆ ಚಿತ್ತವಂತು ಕೆಡುಕಿನ ಕಡೆಗೇ ಪ್ರೇರೇಪಿಸುತ್ತದೆ. ನಿಶ್ಚಯವಾಗಿಯೂ ನನ್ನ ಪ್ರಭು ಕ್ಷಮಾಶೀಲನೂ ಕರುಣಾನಿಧಿಯೂ ಆಗಿರುತ್ತಾನೆ."
12:54 - 55
وَقَالَ ٱلْمَلِكُ ٱئْتُونِى بِهِۦٓ أَسْتَخْلِصْهُ لِنَفْسِى ۖ فَلَمَّا كَلَّمَهُۥ قَالَ إِنَّكَ ٱلْيَوْمَ لَدَيْنَا مَكِينٌ أَمِينٌ٤٥قَالَ ٱجْعَلْنِى عَلَىٰ خَزَآئِنِ ٱلْأَرْضِ ۖ إِنِّى حَفِيظٌ عَلِيمٌ٥٥
ಅರಸನು ಹೇಳಿದನು, "ಅವನನ್ನು ನನ್ನ ಬಳಿಗೆ ಕರೆತನ್ನಿರಿ. ನಾನು ಅವನನ್ನು ನನಗಾಗಿ ಕಾಯ್ದಿರಿಸಿಕೊಳ್ಳುತ್ತೇನೆ." ಯೂಸುಫರು ಅವನೊಡನೆ ಮಾತುಕತೆ ನಡೆಸಿದಾಗ ಅವನು ಹೀಗೆಂದನು, "ಇಂದು ನೀವು ನಮ್ಮಲ್ಲಿ ಗೌರವಾನ್ವಿತರಾಗಿರುತ್ತೀರಿ ಮತ್ತು ನಿಮ್ಮ ಪ್ರಾಮಾಣಿಕತೆಯ ಬಗ್ಗೆ ಸಂಪೂರ್ಣ ಭರವಸೆಯಿದೆ."47 ಯೂಸುಫ್ ಹೇಳಿದರು, "ದೇಶದ ಭಂಡಾರಗಳನ್ನು ನನ್ನ ವಶಕ್ಕೆ ಬಿಡಿರಿ. ನಿಶ್ಚಯವಾಗಿಯೂ ನಾನು ಸಂರಕ್ಷಕ ಮತ್ತು ಜ್ಞಾನಿಯಾಗಿದ್ದೇನೆ."47(ಅ)
12:56 - 57
وَكَذَٰلِكَ مَكَّنَّا لِيُوسُفَ فِى ٱلْأَرْضِ يَتَبَوَّأُ مِنْهَا حَيْثُ يَشَآءُ ۚ نُصِيبُ بِرَحْمَتِنَا مَن نَّشَآءُ ۖ وَلَا نُضِيعُ أَجْرَ ٱلْمُحْسِنِينَ٦٥وَلَأَجْرُ ٱلْـَٔاخِرَةِ خَيْرٌ لِّلَّذِينَ ءَامَنُوا۟ وَكَانُوا۟ يَتَّقُونَ٧٥
ಹೀಗೆ ನಾವು ಯೂಸುಫರಿಗೆ ಆ ಭೂಭಾಗದ ಅಧಿಕಾರವನ್ನು ಸುಗಮಗೊಳಿಸಿದೆವು. ಅಲ್ಲಿ ತನಗೆ ಸರಿಕಂಡಲ್ಲೆಲ್ಲ ಸ್ಥಳ ಮಾಡಿಕೊಳ್ಳಲು ಅವರು ಅಧಿಕಾರ ಹೊಂದಿದ್ದರು.48 ನಾವು ನಮಗಿಷ್ಟವಿದ್ದವರಿಗೆ ನಮ್ಮ ಕೃಪೆಯನ್ನು ಕರುಣಿಸುತ್ತೇವೆ. ನಮ್ಮಲ್ಲಿ ಸದ್ಭಕ್ತರ ಪ್ರತಿಫಲವು ನಷ್ಟಗೊಳಿಸಲ್ಪಡುವುದಿಲ್ಲ. ಸತ್ಯವಿಶ್ವಾಸವನ್ನಿರಿಸಿ, ದೇವಭಯದೊಂದಿಗೆ ಕಾರ್ಯಪ್ರವೃತ್ತರಾಗುವವರಿಗೆ ಪರಲೋಕದ ಪ್ರತಿಫಲವು ಇನ್ನಷ್ಟು ಉತ್ತಮವಾಗಿರುತ್ತದೆ.49
12:58 - 62
وَجَآءَ إِخْوَةُ يُوسُفَ فَدَخَلُوا۟ عَلَيْهِ فَعَرَفَهُمْ وَهُمْ لَهُۥ مُنكِرُونَ٨٥وَلَمَّا جَهَّزَهُم بِجَهَازِهِمْ قَالَ ٱئْتُونِى بِأَخٍ لَّكُم مِّنْ أَبِيكُمْ ۚ أَلَا تَرَوْنَ أَنِّىٓ أُوفِى ٱلْكَيْلَ وَأَنَا۠ خَيْرُ ٱلْمُنزِلِينَ٩٥فَإِن لَّمْ تَأْتُونِى بِهِۦ فَلَا كَيْلَ لَكُمْ عِندِى وَلَا تَقْرَبُونِ٠٦قَالُوا۟ سَنُرَٰوِدُ عَنْهُ أَبَاهُ وَإِنَّا لَفَـٰعِلُونَ١٦وَقَالَ لِفِتْيَـٰنِهِ ٱجْعَلُوا۟ بِضَـٰعَتَهُمْ فِى رِحَالِهِمْ لَعَلَّهُمْ يَعْرِفُونَهَآ إِذَا ٱنقَلَبُوٓا۟ إِلَىٰٓ أَهْلِهِمْ لَعَلَّهُمْ يَرْجِعُونَ٢٦
ಯೂಸುಫರ ಸಹೋದರರು ಈಜಿಪ್ಟಿಗೆ ಬಂದರು ಮತ್ತು ಇವರ ಮುಂದೆ ಹಾಜರಾದರು.50 ಇವರು ಅವರನ್ನು ಗುರುತಿಸಿದರು. ಆದರೆ ಅವರಿಗೆ ಇವರ ಗುರುತು ಸಿಗಲಿಲ್ಲ.51 ಇವರು ಅವರ ಸರಕುಗಳನ್ನೆಲ್ಲಾ ಸಿದ್ಧಗೊಳಿಸಿದ ಬಳಿಕ ಹೊರಡುವಾಗ ಹೀಗೆ ಹೇಳಿದರು, "ನಿಮ್ಮ ಮಲ ಸಹೋದರನನ್ನು ನನ್ನ ಬಳಿಗೆ ಕರೆ ತರಬೇಕು. ನಾನು ಹೇಗೆ ಅಳತೆ ಪಾತ್ರೆ ತುಂಬಿ ಕೊಡುತ್ತೇನೆಂದೂ ಎಷ್ಟು ಉತ್ತಮ ಆತಿಥೇಯನಾಗಿದ್ದೇನೆಂದೂ ಕಾಣುತ್ತೀರಲ್ಲವೆ? ನೀವು ಅವನನ್ನು ಕರೆತರದಿದ್ದರೆ ನನ್ನಲ್ಲಿ ನಿಮಗೆ ಧಾನ್ಯವಿಲ್ಲ. ಮಾತ್ರವಲ್ಲ ನೀವು ನನ್ನ ಬಳಿ ಸುಳಿಯಲೂ ಬಾರದು."52 ಅವರು ಹೇಳಿದರು, "ತಂದೆಯವರು ಅವನನ್ನು ಕಳುಹಿಸಲು ಒಪ್ಪುವಂತೆ ಮಾಡಲು ಪ್ರಯತ್ನಿಸುವೆವು. ಹೀಗೆ ನಾವು ಮಾಡಿಯೇ ತೀರುವೆವು." ಧಾನ್ಯದ ಪ್ರತಿಯಾಗಿ ಅವರು ಕೊಟ್ಟಿದ್ದ ಮೊಬಲಗನ್ನು ಅವರ ಸರಕಿನಲ್ಲೇ ಗುಪ್ತವಾಗಿ ಇರಿಸಿಬಿಡಿರೆಂದು ಯೂಸುಫರು ತನ್ನ ಗುಲಾಮರೊಡನೆ ಹೇಳಿದರು. ಅವರು ಮನೆಗೆ ತಲುಪಿದಾಗ ತಮಗೆ ಮರಳಿ ಸಿಕ್ಕಿದ ಮೊಬಲಗನ್ನು ಗುರುತಿಸಿಕೊಳ್ಳುವರು (ಅಥವಾ ಈ ಔದಾರ್ಯಕ್ಕಾಗಿ ಕೃತಜ್ಞರಾಗುವರು) ಮತ್ತು ಪುನಃ ಮರಳಿ ಬರಬಹುದೆಂಬ ನಿರೀಕ್ಷೆಯಿಂದ ಯೂಸುಫ್ ಹೀಗೆ ಮಾಡಿದ್ದರು.
12:63 - 68
فَلَمَّا رَجَعُوٓا۟ إِلَىٰٓ أَبِيهِمْ قَالُوا۟ يَـٰٓأَبَانَا مُنِعَ مِنَّا ٱلْكَيْلُ فَأَرْسِلْ مَعَنَآ أَخَانَا نَكْتَلْ وَإِنَّا لَهُۥ لَحَـٰفِظُونَ٣٦قَالَ هَلْ ءَامَنُكُمْ عَلَيْهِ إِلَّا كَمَآ أَمِنتُكُمْ عَلَىٰٓ أَخِيهِ مِن قَبْلُ ۖ فَٱللَّهُ خَيْرٌ حَـٰفِظًا ۖ وَهُوَ أَرْحَمُ ٱلرَّٰحِمِينَ٤٦وَلَمَّا فَتَحُوا۟ مَتَـٰعَهُمْ وَجَدُوا۟ بِضَـٰعَتَهُمْ رُدَّتْ إِلَيْهِمْ ۖ قَالُوا۟ يَـٰٓأَبَانَا مَا نَبْغِى ۖ هَـٰذِهِۦ بِضَـٰعَتُنَا رُدَّتْ إِلَيْنَا ۖ وَنَمِيرُ أَهْلَنَا وَنَحْفَظُ أَخَانَا وَنَزْدَادُ كَيْلَ بَعِيرٍ ۖ ذَٰلِكَ كَيْلٌ يَسِيرٌ٥٦قَالَ لَنْ أُرْسِلَهُۥ مَعَكُمْ حَتَّىٰ تُؤْتُونِ مَوْثِقًا مِّنَ ٱللَّهِ لَتَأْتُنَّنِى بِهِۦٓ إِلَّآ أَن يُحَاطَ بِكُمْ ۖ فَلَمَّآ ءَاتَوْهُ مَوْثِقَهُمْ قَالَ ٱللَّهُ عَلَىٰ مَا نَقُولُ وَكِيلٌ٦٦وَقَالَ يَـٰبَنِىَّ لَا تَدْخُلُوا۟ مِنۢ بَابٍ وَٰحِدٍ وَٱدْخُلُوا۟ مِنْ أَبْوَٰبٍ مُّتَفَرِّقَةٍ ۖ وَمَآ أُغْنِى عَنكُم مِّنَ ٱللَّهِ مِن شَىْءٍ ۖ إِنِ ٱلْحُكْمُ إِلَّا لِلَّهِ ۖ عَلَيْهِ تَوَكَّلْتُ ۖ وَعَلَيْهِ فَلْيَتَوَكَّلِ ٱلْمُتَوَكِّلُونَ٧٦وَلَمَّا دَخَلُوا۟ مِنْ حَيْثُ أَمَرَهُمْ أَبُوهُم مَّا كَانَ يُغْنِى عَنْهُم مِّنَ ٱللَّهِ مِن شَىْءٍ إِلَّا حَاجَةً فِى نَفْسِ يَعْقُوبَ قَضَىٰهَا ۚ وَإِنَّهُۥ لَذُو عِلْمٍ لِّمَا عَلَّمْنَـٰهُ وَلَـٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ٨٦
ಅವರು ತಮ್ಮ ತಂದೆಯವರ ಬಳಿಗೆ ಹೋದಾಗ ಹೀಗೆಂದರು, "ಅಪ್ಪಾ, ಇನ್ನು ಮುಂದೆ ನಮಗೆ ಧಾನ್ಯ ಕೊಡಲು ನಿರಾಕರಿಸಲಾಗಿದೆ. ಆದುದರಿಂದ ನಾವು ಧಾನ್ಯವನ್ನು ತರಲು ನೀವು ನಮ್ಮ ಸಂಗಡ ನಮ್ಮ ಸಹೋದರನನ್ನು ಕಳುಹಿಸಿ ಕೊಡಿರಿ. ಅವನ ರಕ್ಷಣೆಗೆ ನಾವು ಹೊಣೆಗಾರರು." ಆಗ ತಂದೆಯು, "ನಾನು ಇದಕ್ಕೆ ಮುಂಚೆ ಇವನ ಅಣ್ಣನ ವಿಷಯದಲ್ಲಿ ನಿಮ್ಮ ಮೇಲೆ ಭರವಸೆಯನ್ನಿರಿಸಿದ್ದಂತೆಯೇ ಇವನ ವಿಷಯದಲ್ಲೂ ಇಡಬೇಕೇ? ಅಲ್ಲಾಹನೇ ಅತ್ಯುತ್ತಮ ರಕ್ಷಕನು ಮತ್ತು ಅವನು ಎಲ್ಲರಿಗಿಂತಲೂ ಹೆಚ್ಚು ಕರುಣಾಳುವಾಗಿರುತ್ತಾನೆ" ಎಂದರು. ತರುವಾಯ ಅವರು ತಮ್ಮ ಸರಕನ್ನು ಬಿಚ್ಚಿದಾಗ ಅವರ ಮೊಬಲಗು ಅವರಿಗೆ ಹಿಂದಿರುಗಿಸಲಾಗಿರುವುದನ್ನು ಕಂಡು, "ಅಪ್ಪಾ, ನಮಗೆ ಇನ್ನೇನು ಬೇಕು? ಇಲ್ಲಿ ನೋಡಿರಿ. ನಮ್ಮ ಹಣವನ್ನೂ ನಮಗೆ ಹಿಂದಿರುಗಿಸಲಾಗಿದೆ. ನಾವಿನ್ನು ಹೋಗುವೆವು. ನಮ್ಮ ಕುಟುಂಬಕ್ಕಾಗಿ ಆಹಾರ ಧಾನ್ಯ ತರುವೆವು. ನಮ್ಮ ಸಹೋದರನ ರಕ್ಷಣೆಯನ್ನೂ ಮಾಡುವೆವು ಮತ್ತು ಒಂದು ಒಂಟೆ ಹೊರೆಯನ್ನು ಹೆಚ್ಚುವರಿಯಾಗಿ ತರುವೆವು. ಇಷ್ಟು ಧಾನ್ಯದ ಹೆಚ್ಚಳವು ಸುಲಭವಾಗಿ ಆಗಿಬಿಡುವುದು" ಎಂದು ಉದ್ಗರಿಸಿದರು. ಅವರ ತಂದೆ ಹೇಳಿದರು- "ನೀವೇ ಆಕ್ರಮಣಕ್ಕೊಳಗಾಗಿ ಬಿಡುವುದರ ಹೊರತು ಅವನನ್ನು ನನ್ನ ಬಳಿಗೆ ಮರಳಿ ಕರೆತರುವಿರೆಂದು ನೀವು ನನಗೆ ಅಲ್ಲಾಹನ ಹೆಸರಲ್ಲಿ ಪ್ರಮಾಣ ವಚನ ನೀಡುವ ತನಕ ನಾನು ಅವನನ್ನು ನಿಮ್ಮ ಸಂಗಡ ಸರ್ವಥಾ ಕಳುಹಿಸಿ ಕೊಡಲಾರೆ." ಅವರಿಗೆ (ಹ. ಯಾ‌ಕೂಬರಿಗೆ) ಅವರೆಲ್ಲರೂ ತಮ್ಮ ತಮ್ಮ ಪ್ರಮಾಣವಚನ ಕೊಟ್ಟ ಬಳಿಕ ಅವರು (ಯಾ‌ಕೂಬರು) ಹೇಳಿದರು, "ನಮ್ಮ ಈ ಪ್ರತಿಜ್ಞೆಯ ಮೇಲ್ನೋಟವಿರಿಸುವವನು ಅಲ್ಲಾಹ್." ತರುವಾಯ ಅವರು ಹೇಳಿದರು, "ಓ ನನ್ನ ಮಕ್ಕಳೇ, ನೀವು ಈಜಿಪ್ಟಿನ ರಾಜಧಾನಿಯೊಳಗೆ ಒಂದೇ ದ್ವಾರದಿಂದ ಪ್ರವೇಶಿಸಬಾರದು. ಬೇರೆ ಬೇರೆ ದ್ವಾರಗಳಿಂದ ಪ್ರವೇಶಿಸಬೇಕು.53 ಆದರೆ ನಾನು ನಿಮ್ಮನ್ನು ಅಲ್ಲಾಹನ ವಿಧಿಯಿಂದ ತಪ್ಪಿಸಲಾರೆ. ಆಜ್ಞಾಧಿಕಾರವು ಅವನ ಹೊರತು ಬೇರೆ ಯಾರಿಗೂ ಇಲ್ಲ. ನಾನು ಅವನ ಮೇಲೆಯೇ ಭರವಸೆಯನ್ನಿರಿಸಿದೆ ಮತ್ತು ಯಾರು ಭರವಸೆಯನ್ನಿರಿಸುವುದಿದ್ದರೂ ಅವನ ಮೇಲೆಯೇ ಇರಿಸಲಿ." ವಾಸ್ತವದಲ್ಲಿ ಆದುದೂ ಇದೇ. ಅವರು ತಮ್ಮ ತಂದೆಯ ಆದೇಶದಂತೆ (ಬೇರೆ ಬೇರೆ ದ್ವಾರಗಳಿಂದ) ನಗರದೊಳಗೆ ಪ್ರವೇಶಿಸಿದ್ದ ಈ ಮುಂಜಾಗ್ರತಾ ಯತ್ನವು ಅಲ್ಲಾಹನ ವಿಧಿಯ ಮುಂದೆ ಏನೂ ಫಲಕಾರಿಯಾಗಲಿಲ್ಲ. ಯಾ‌ಕೂಬರು ತಮ್ಮ ಮನಸ್ಸಿನಲ್ಲಿದ್ದ ಒಂದು ಅಂಜಿಕೆಯನ್ನು ದೂರೀಕರಿಸಲಿಕ್ಕಾಗಿ ಅವರು ತಮ್ಮ ಪ್ರಯತ್ನ ಮಾಡಿಕೊಂಡರಷ್ಟೆ. ನಿಸ್ಸಂದೇಹವಾಗಿಯೂ ನಾವು ನೀಡಿದ ಶಿಕ್ಷಣದಿಂದ ಅವರು ಸುಶಿಕ್ಷಿತರಾಗಿಯೇ ಇದ್ದರು. ಆದರೆ ಹೆಚ್ಚಿನವರು ಪರಮಾರ್ಥವನ್ನು ತಿಳಿಯುವುದಿಲ್ಲ.54
12:69 - 69
وَلَمَّا دَخَلُوا۟ عَلَىٰ يُوسُفَ ءَاوَىٰٓ إِلَيْهِ أَخَاهُ ۖ قَالَ إِنِّىٓ أَنَا۠ أَخُوكَ فَلَا تَبْتَئِسْ بِمَا كَانُوا۟ يَعْمَلُونَ٩٦
ಇವರು ಯೂಸುಫರ ಸನ್ನಿಧಿಗೆ ತಲುಪಿದಾಗ ಅವರು ತಮ್ಮನನ್ನು ಒತ್ತಟ್ಟಿಗೆ ಕರೆದು, "(ಕಾಣೆಯಾಗಿದ್ದ) ನಿನ್ನ ಆ ಸಹೋದರನೇ ನಾನು. ಇನ್ನು ನೀನು ಇವರು ವರ್ತಿಸುತ್ತಿದ್ದುದರ ಬಗ್ಗೆ ಚಿಂತಿಸಬೇಡ" ಎಂದರು.55
12:70 - 76
فَلَمَّا جَهَّزَهُم بِجَهَازِهِمْ جَعَلَ ٱلسِّقَايَةَ فِى رَحْلِ أَخِيهِ ثُمَّ أَذَّنَ مُؤَذِّنٌ أَيَّتُهَا ٱلْعِيرُ إِنَّكُمْ لَسَـٰرِقُونَ٠٧قَالُوا۟ وَأَقْبَلُوا۟ عَلَيْهِم مَّاذَا تَفْقِدُونَ١٧قَالُوا۟ نَفْقِدُ صُوَاعَ ٱلْمَلِكِ وَلِمَن جَآءَ بِهِۦ حِمْلُ بَعِيرٍ وَأَنَا۠ بِهِۦ زَعِيمٌ٢٧قَالُوا۟ تَٱللَّهِ لَقَدْ عَلِمْتُم مَّا جِئْنَا لِنُفْسِدَ فِى ٱلْأَرْضِ وَمَا كُنَّا سَـٰرِقِينَ٣٧قَالُوا۟ فَمَا جَزَٰٓؤُهُۥٓ إِن كُنتُمْ كَـٰذِبِينَ٤٧قَالُوا۟ جَزَٰٓؤُهُۥ مَن وُجِدَ فِى رَحْلِهِۦ فَهُوَ جَزَٰٓؤُهُۥ ۚ كَذَٰلِكَ نَجْزِى ٱلظَّـٰلِمِينَ٥٧فَبَدَأَ بِأَوْعِيَتِهِمْ قَبْلَ وِعَآءِ أَخِيهِ ثُمَّ ٱسْتَخْرَجَهَا مِن وِعَآءِ أَخِيهِ ۚ كَذَٰلِكَ كِدْنَا لِيُوسُفَ ۖ مَا كَانَ لِيَأْخُذَ أَخَاهُ فِى دِينِ ٱلْمَلِكِ إِلَّآ أَن يَشَآءَ ٱللَّهُ ۚ نَرْفَعُ دَرَجَـٰتٍ مَّن نَّشَآءُ ۗ وَفَوْقَ كُلِّ ذِى عِلْمٍ عَلِيمٌ٦٧
ಯೂಸುಫರು ತಮ್ಮ ಸಹೋದರರ ಸರಕನ್ನು ಹೇರಿಸುತ್ತಿರುವಾಗ ತಮ್ಮ ತಮ್ಮನ ಸರಕಿನಲ್ಲಿ ಲೋಟೆಯನ್ನು ಇಟ್ಟು ಬಿಟ್ಟರು.56 ತರುವಾಯ ಕೂಗುವವನೊಬ್ಬನು, "ಓ ಯಾತ್ರಿಕ ತಂಡದವರೇ, ನೀವು ಕಳ್ಳರಾಗಿರುತ್ತೀರಿ" ಎಂದು ಕೂಗಿದನು.57 ಅವರು ತಿರುಗಿ, "ನಿಮ್ಮದೇನು ಕಳೆದುಹೋಗಿದೆ?" ಎಂದು ಕೇಳಿದರು. ಆಗ ಸರಕಾರಿ ನೌಕರರು, "ರಾಜನ ಲೋಟೆಯು ಕಳೆದುಹೋಗಿದೆ" ಎಂದರು. (ಅವರ ನಾಯಕನು) "ಅದನ್ನು ತಂದು ಕೊಟ್ಟವನಿಗೆ ಒಂದು ಒಂಟೆ ಹೊರುವಷ್ಟು ಧಾನ್ಯ ಬಹುಮಾನವಿದೆ. ಇದಕ್ಕೆ ನಾನು ಹೊಣೆಗಾರನಾಗಿದ್ದೇನೆ" ಎಂದನು. ಆ ಸಹೋದರರು, "ಅಲ್ಲಾಹನಾಣೆ! ನಾವು ಈ ದೇಶದಲ್ಲಿ ಗೊಂದಲವನ್ನುಂಟು ಮಾಡಲು ಬಂದವರಲ್ಲ ಮತ್ತು ನಾವು ಕಳ್ಳತನ ಮಾಡುವವರಲ್ಲವೆಂದು ನೀವು ಚೆನ್ನಾಗಿ ಬಲ್ಲಿರಿ" ಎಂದರು. "ಸರಿ, ನೀವು ಸುಳ್ಳುಗಾರರಾದರೆ ಕಳ್ಳನಿಗೆ ಶಿಕ್ಷೆಯೇನು?" ಎಂದು ಅವರು ಕೇಳಿದರು. ಅದಕ್ಕೆ ಅವರು, "ಅವನಿಗೆ ಶಿಕ್ಷೆಯೇ? ಯಾರ ಸರಕಿನೊಳಗಿಂದ ಅದು ಸಿಗುವುದೋ ಅವನೇ ಅದರ ಬದಲಿಗೆ ಇರಿಸಲ್ಪಡಬೇಕು. ನಮ್ಮಲ್ಲಿ ಇಂತಹ ಅಕ್ರಮಿಗಳಿಗೆ ಶಿಕ್ಷೆ ಕೊಡುವ ಕ್ರಮ ಇದುವೇ" ಎಂದು ಹೇಳಿದರು.58 ಆಗ ಯೂಸುಫ್ ತನ್ನ ಸಹೋದರನ ಚೀಲಕ್ಕಿಂತ ಮುಂಚೆ ಅವರ ಚೀಲಗಳ ತಪಾಸಣೆ ನಡೆಸಲಾರಂಭಿಸಿದರು. ಆ ಬಳಿಕ ತನ್ನ ತಮ್ಮನ ಚೀಲದಿಂದ, ಕಾಣೆಯಾಗಿದ್ದುದನ್ನು ಹೊರ ತೆಗೆದರು. ಹೀಗೆ ನಾವು ನಮ್ಮ ಯುಕ್ತಿಯಿಂದ ಯೂಸುಫರನ್ನು ಸಮರ್ಥಿಸಿದೆವು.59 ಅಲ್ಲಾಹನೇ ಇಚ್ಛಿಸಿದ ವಿನಾ ರಾಜಧರ್ಮ (ಅರ್ಥಾತ್ ಈಜಿಪ್ಟಿನ ರಾಜ ಶಾಸನದ) ಪ್ರಕಾರ ತನ್ನ ತಮ್ಮನನ್ನು ಹಿಡಿಯುವುದು ಅವರಿಗೆ ಯೋಗ್ಯವಾಗಿರಲಿಲ್ಲ.60 ನಾವು ನಮಗಿಷ್ಟ ಬಂದವರ ಸ್ಥಾನವನ್ನು ಉನ್ನತಗೊಳಿಸುತ್ತೇವೆ. ಸಕಲ ಜ್ಞಾನಿಗಳಿಗಿಂತ ಮೇಲಾಗಿರುವ ಓರ್ವ ಸರ್ವಜ್ಞನಿದ್ದಾನೆ.
12:77 - 77
۞ قَالُوٓا۟ إِن يَسْرِقْ فَقَدْ سَرَقَ أَخٌ لَّهُۥ مِن قَبْلُ ۚ فَأَسَرَّهَا يُوسُفُ فِى نَفْسِهِۦ وَلَمْ يُبْدِهَا لَهُمْ ۚ قَالَ أَنتُمْ شَرٌّ مَّكَانًا ۖ وَٱللَّهُ أَعْلَمُ بِمَا تَصِفُونَ٧٧
ಆ ಸಹೋದರರು ಹೇಳಿದರು, "ಈತನು ಕದ್ದರೆ ಆಶ್ಚರ್ಯವಿಲ್ಲ. ಇದಕ್ಕಿಂತ ಮುಂಚೆ ಇವನ ಅಣ್ಣನೂ (ಯೂಸುಫ್) ಕಳವು ಮಾಡಿದ್ದಾನೆ."61 ಯೂಸುಫ್, ಅವರ ಈ ಮಾತನ್ನು ನುಂಗಿಕೊಂಡರು. ನಿಜಸ್ಥಿತಿಯನ್ನು ಬಹಿರಂಗಗೊಳಿಸಲಿಲ್ಲ. "ಮಹಾ ನೀಚರು ನೀವು! (ನನ್ನ ಸಮ್ಮುಖದಲ್ಲೇ) ನೀವು ಹೊರಿಸುತ್ತಿರುವ ಅಪವಾದದ ನಿಜಸ್ಥಿತಿಯನ್ನು ಅಲ್ಲಾಹನು ಚೆನ್ನಾಗಿ ಬಲ್ಲವನಾಗಿರುತ್ತಾನೆ" ಎಂದಿಷ್ಟೆ (ಸ್ವಗತ) ಹೇಳಿಕೊಂಡು ಸುಮ್ಮನಾದರು.
12:78 - 79
قَالُوا۟ يَـٰٓأَيُّهَا ٱلْعَزِيزُ إِنَّ لَهُۥٓ أَبًا شَيْخًا كَبِيرًا فَخُذْ أَحَدَنَا مَكَانَهُۥٓ ۖ إِنَّا نَرَىٰكَ مِنَ ٱلْمُحْسِنِينَ٨٧قَالَ مَعَاذَ ٱللَّهِ أَن نَّأْخُذَ إِلَّا مَن وَجَدْنَا مَتَـٰعَنَا عِندَهُۥٓ إِنَّآ إِذًا لَّظَـٰلِمُونَ٩٧
ಅವರು ಹೇಳಿದರು, "ಓ ಅಧಿಕಾರಿಯೇ, (ಅಝೀಝ್!)62 ಇವನ ತಂದೆ ತುಂಬಾ ವಯೋವೃದ್ಧರು. ನೀವು ಇವನ ಸ್ಥಾನದಲ್ಲಿ ನಮ್ಮ ಪೈಕಿ ಯಾರನ್ನಾದರೂ ತಡೆದಿರಿಸಿಕೊಳ್ಳಿರಿ. ನಾವು ನಿಮ್ಮನ್ನು ಮಹಾ ಸಜ್ಜನರನ್ನಾಗಿ ಕಾಣುತ್ತೇವೆ." ಆಗ ಯೂಸುಫ್ ಹೇಳಿದರು, "ಶಾಂತಂ ಪಾಪಂ. ಬೇರೊಬ್ಬನನ್ನು ನಾವು ಇರಿಸಿಕೊಳ್ಳುವುದಾದರೂ ಹೇಗೆ? ನಾವು ನಮ್ಮ ಸೊತ್ತನ್ನು ಯಾರಲ್ಲಿ ಕಂಡೆವೋ63 ಅವನನ್ನು ಬಿಟ್ಟು ಬೇರೊಬ್ಬನನ್ನು ಇರಿಸಿಕೊಂಡರೆ ನಾವು ಅಕ್ರಮಿಗಳಾಗುವೆವು."
12:80 - 82
فَلَمَّا ٱسْتَيْـَٔسُوا۟ مِنْهُ خَلَصُوا۟ نَجِيًّا ۖ قَالَ كَبِيرُهُمْ أَلَمْ تَعْلَمُوٓا۟ أَنَّ أَبَاكُمْ قَدْ أَخَذَ عَلَيْكُم مَّوْثِقًا مِّنَ ٱللَّهِ وَمِن قَبْلُ مَا فَرَّطتُمْ فِى يُوسُفَ ۖ فَلَنْ أَبْرَحَ ٱلْأَرْضَ حَتَّىٰ يَأْذَنَ لِىٓ أَبِىٓ أَوْ يَحْكُمَ ٱللَّهُ لِى ۖ وَهُوَ خَيْرُ ٱلْحَـٰكِمِينَ٠٨ٱرْجِعُوٓا۟ إِلَىٰٓ أَبِيكُمْ فَقُولُوا۟ يَـٰٓأَبَانَآ إِنَّ ٱبْنَكَ سَرَقَ وَمَا شَهِدْنَآ إِلَّا بِمَا عَلِمْنَا وَمَا كُنَّا لِلْغَيْبِ حَـٰفِظِينَ١٨وَسْـَٔلِ ٱلْقَرْيَةَ ٱلَّتِى كُنَّا فِيهَا وَٱلْعِيرَ ٱلَّتِىٓ أَقْبَلْنَا فِيهَا ۖ وَإِنَّا لَصَـٰدِقُونَ٢٨
ಹೀಗೆ ಯೂಸುಫರಿಂದ ನಿರಾಶರಾದ ಮೇಲೆ ಅವರು ಒತ್ತಟ್ಟಿಗೆ ಹೋಗಿ ಪರಸ್ಪರ ಸಮಾಲೋಚನೆ ನಡೆಸತೊಡಗಿದರು. ಅವರಲ್ಲಿ ಎಲ್ಲರಿಗೂ ಹಿರಿಯನಾಗಿದ್ದವನು ಹೇಳಿದನು, "ನಿಮ್ಮ ತಂದೆ ನಿಮ್ಮೊಡನೆ ಅಲ್ಲಾಹನ ಹೆಸರಲ್ಲಿ ಪ್ರಮಾಣ ವಚನ ಪಡೆದಿರುವುದು ನಿಮಗೆ ತಿಳಿದಿಲ್ಲವೇ? ಇದಕ್ಕೆ ಮುಂಚೆ ಯೂಸುಫನ ವಿಷಯದಲ್ಲಿ ನೀವು ಮಾಡಿರುವುದೆಲ್ಲವೂ ನಿಮಗೆ ತಿಳಿದಿದೆ. ಇನ್ನು ನನ್ನ ತಂದೆಯವರು ನನಗೆ ಅನುಮತಿ ಕೊಡುವವರೆಗೆ ಅಥವಾ ಅಲ್ಲಾಹನೇ ನನ್ನ ಬಗ್ಗೆ ತೀರ್ಮಾನ ಮಾಡುವವರೆಗೆ ನಾನು ಇಲ್ಲಿಂದ ಎಷ್ಟು ಮಾತ್ರಕ್ಕೂ ಹೋಗಲಾರೆ. ಅಲ್ಲಾಹ್ ಅತ್ಯುತ್ತಮ ತೀರ್ಪುಗಾರನು. ನೀವು ಮರಳಿ ಹೋಗಿ ತಂದೆಯವರೊಡನೆ, ಅಪ್ಪಾ, ನಿಮ್ಮ ಮಗನು ಕಳವು ಮಾಡಿದ್ದಾನೆ. ಅವನು ಕಳವು ಮಾಡುತ್ತಿರುವುದನ್ನು ನಾವು ನೋಡಲಿಲ್ಲ. ನಮಗೆ ತಿಳಿದುದನ್ನು ಮಾತ್ರ ನಾವು ಹೇಳುತ್ತೇವೆ. ಅದೃಶ್ಯವಾಗಿರುವುದರ ಬಗ್ಗೆ ಮೇಲ್ನೋಟ ವಹಿಸಲು ನಮ್ಮಿಂದ ಸಾಧ್ಯವಿರಲಿಲ್ಲ. ತಾವು ನಾವಿದ್ದ ನಾಡಿನ ಜನರೊಡನೆಯೂ ನಾವು ಯಾವ ಯಾತ್ರಾ ತಂಡದೊಂದಿಗೆ ಬಂದಿದ್ದೇವೋ ಅವರೊಡನೆಯೂ ಕೇಳಿಕೊಳ್ಳಿರಿ. ನಾವು ಸಂಪೂರ್ಣ ಸತ್ಯವಾದಿಗಳು ಎಂದು ಹೇಳಿರಿ" ಎಂದನು.

قَالَ بَلْ سَوَّلَتْ لَكُمْ أَنفُسُكُمْ أَمْرًا ۖ فَصَبْرٌ جَمِيلٌ ۖ عَسَى ٱللَّهُ أَن يَأْتِيَنِى بِهِمْ جَمِيعًا ۚ إِنَّهُۥ هُوَ ٱلْعَلِيمُ ٱلْحَكِيمُ٣٨وَتَوَلَّىٰ عَنْهُمْ وَقَالَ يَـٰٓأَسَفَىٰ عَلَىٰ يُوسُفَ وَٱبْيَضَّتْ عَيْنَاهُ مِنَ ٱلْحُزْنِ فَهُوَ كَظِيمٌ٤٨قَالُوا۟ تَٱللَّهِ تَفْتَؤُا۟ تَذْكُرُ يُوسُفَ حَتَّىٰ تَكُونَ حَرَضًا أَوْ تَكُونَ مِنَ ٱلْهَـٰلِكِينَ٥٨قَالَ إِنَّمَآ أَشْكُوا۟ بَثِّى وَحُزْنِىٓ إِلَى ٱللَّهِ وَأَعْلَمُ مِنَ ٱللَّهِ مَا لَا تَعْلَمُونَ٦٨يَـٰبَنِىَّ ٱذْهَبُوا۟ فَتَحَسَّسُوا۟ مِن يُوسُفَ وَأَخِيهِ وَلَا تَا۟يْـَٔسُوا۟ مِن رَّوْحِ ٱللَّهِ ۖ إِنَّهُۥ لَا يَا۟يْـَٔسُ مِن رَّوْحِ ٱللَّهِ إِلَّا ٱلْقَوْمُ ٱلْكَـٰفِرُونَ٧٨
ಈ ವೃತ್ತಾಂತವನ್ನು ಕೇಳಿ ತಂದೆಯು, "ವಾಸ್ತವದಲ್ಲಿ ನಿಮ್ಮ ಭ್ರಷ್ಟಚಿತ್ತವು ನಿಮಗೆ ಇನ್ನೊಂದು ದೊಡ್ಡ ವಿಷಯವನ್ನು ಸುಲಭಗೊಳಿಸಿಬಿಟ್ಟಿತು.64 ಚಿಂತಿಲ್ಲ; ಇದನ್ನೂ ತಾಳಿಕೊಳ್ಳುವೆನು; ಚೆನ್ನಾಗಿ ತಾಳಿಕೊಳ್ಳುವೆನು. ಅಲ್ಲಾಹನು ಅವರೆಲ್ಲರನ್ನೂ ತಂದು ನನ್ನೊಂದಿಗೆ ಸೇರಿಸಿಬಿಡಲೂ ಬಹುದು. ನಿಶ್ಚಯವಾಗಿಯೂ ಅಲ್ಲಾಹನು ಸರ್ವಜ್ಞನಾಗಿದ್ದು ಅವನ ಎಲ್ಲಾ ಕಾರ್ಯಗಳೂ ಯುಕ್ತಿಪೂರ್ಣವಾಗಿರುತ್ತವೆ" ಎಂದು ಹೇಳಿದರು. ಅನಂತರ ಅವರು (ಯಾ‌ಕೂಬರು) ಅವರಿಂದ ಮುಖ ತಿರುಗಿಸಿ ಕುಳಿತರು ಮತ್ತು 'ಹಾ! ಯೂಸುಫ್!' ಎಂದು ಪ್ರಲಾಪಿಸತೊಡಗಿದರು. ಅವರು ಒಳಗಿಂದೊಳಗೇ ಶೋಕಾತಿರೇಕದಿಂದ ಏದುಸಿರು ಬಿಡುತ್ತಿದ್ದರು ಮತ್ತು ಅವರ ಕಣ್ಣುಗಳು ಬಿಳುಪೇರಿ ಬಿಟ್ಟಿದ್ದುವು. ಪುತ್ರರು ಹೇಳಿದರು, "ಅಲ್ಲಾಹನಾಣೆ! ತಾವಂತು ಯೂಸುಫನನ್ನೇ ನೆನೆಸುತ್ತ ಅವನ ದುಃಖದಲ್ಲಿ ತಮ್ಮನ್ನು ತಾವೇ ಕರಗಿಸಿಬಿಡುವಿರಿ ಅಥವಾ ತಮ್ಮ ಜೀವವನ್ನೇ ನಾಶಗೊಳಿಸಿಕೊಳ್ಳುವಿರಿ." ಅವರು ಹೀಗೆ ಹೇಳಿದರು: "ನಾನು ನನ್ನ ತಲ್ಲಣ ಮತ್ತು ದುಃಖದ ಮೊರೆಯನ್ನು ಅಲ್ಲಾಹನ ಹೊರತು ಇನ್ನಾರೊಡನೆಯೂ ಇಡುವುದಿಲ್ಲ. ಅಲ್ಲಾಹನನ್ನು ನಾನು ತಿಳಿದುಕೊಂಡಂತೆ ನೀವು ತಿಳಿದಿಲ್ಲ. ಓ ನನ್ನ ಮಕ್ಕಳೇ, ನೀವು ಹೋಗಿ ಯೂಸುಫನನ್ನೂ ಅವನ ತಮ್ಮನನ್ನೂ ಪತ್ತೆಹಚ್ಚಿರಿ. ಅಲ್ಲಾಹನ ಅನುಗ್ರಹದಿಂದ ನಿರಾಶರಾಗಬೇಡಿರಿ. ಅವನ ಅನುಗ್ರಹದಿಂದ ಸತ್ಯನಿಷೇಧಿಗಳು ಮಾತ್ರ ನಿರಾಶರಾಗುತ್ತಾರೆ."
12:88 - 93
فَلَمَّا دَخَلُوا۟ عَلَيْهِ قَالُوا۟ يَـٰٓأَيُّهَا ٱلْعَزِيزُ مَسَّنَا وَأَهْلَنَا ٱلضُّرُّ وَجِئْنَا بِبِضَـٰعَةٍ مُّزْجَىٰةٍ فَأَوْفِ لَنَا ٱلْكَيْلَ وَتَصَدَّقْ عَلَيْنَآ ۖ إِنَّ ٱللَّهَ يَجْزِى ٱلْمُتَصَدِّقِينَ٨٨قَالَ هَلْ عَلِمْتُم مَّا فَعَلْتُم بِيُوسُفَ وَأَخِيهِ إِذْ أَنتُمْ جَـٰهِلُونَ٩٨قَالُوٓا۟ أَءِنَّكَ لَأَنتَ يُوسُفُ ۖ قَالَ أَنَا۠ يُوسُفُ وَهَـٰذَآ أَخِى ۖ قَدْ مَنَّ ٱللَّهُ عَلَيْنَآ ۖ إِنَّهُۥ مَن يَتَّقِ وَيَصْبِرْ فَإِنَّ ٱللَّهَ لَا يُضِيعُ أَجْرَ ٱلْمُحْسِنِينَ٠٩قَالُوا۟ تَٱللَّهِ لَقَدْ ءَاثَرَكَ ٱللَّهُ عَلَيْنَا وَإِن كُنَّا لَخَـٰطِـِٔينَ١٩قَالَ لَا تَثْرِيبَ عَلَيْكُمُ ٱلْيَوْمَ ۖ يَغْفِرُ ٱللَّهُ لَكُمْ ۖ وَهُوَ أَرْحَمُ ٱلرَّٰحِمِينَ٢٩ٱذْهَبُوا۟ بِقَمِيصِى هَـٰذَا فَأَلْقُوهُ عَلَىٰ وَجْهِ أَبِى يَأْتِ بَصِيرًا وَأْتُونِى بِأَهْلِكُمْ أَجْمَعِينَ٣٩
ಇವರು ಈಜಿಪ್ಟಿಗೆ ಹೋಗಿ ಯೂಸುಫರ ಮುಂದೆ ಹಾಜರಾಗಿ ಹೀಗೆ ಬಿನ್ನವಿಸಿದರು, "ಓ ಮಹಾನುಭಾವರೇ, ನಾವು ಮತ್ತು ನಮ್ಮ ಮಕ್ಕಳು ಮರಿಗಳು ಮಹಾ ಗಂಡಾಂತರದಲ್ಲಿ ಸಿಲುಕಿದ್ದೇವೆ. ನಾವು ಅತ್ಯಲ್ಪ ಬಂಡವಾಳ ತಂದಿದ್ದೇವೆ. ನೀವು ನಮಗೆ ಹೇರಳ ಧಾನ್ಯ ನೀಡಿರಿ ಮತ್ತು ನಮಗೆ ದಾನ ಮಾಡಿರಿ.65 ಅಲ್ಲಾಹ್ ದಾನ ಮಾಡುವವರಿಗೆ ಪ್ರತಿಫಲ ನೀಡುತ್ತಾನೆ." (ಇದನ್ನು ಕೇಳಿ ಯೂಸುಫರಿಗೆ ಸಹಿಸಲಾಗಲಿಲ್ಲ) ಅವರು (ಹೀಗೆ) ಹೇಳಿದರು: "ನೀವು ಅಜ್ಞಾನಿಗಳಾಗಿದ್ದಾಗ ಯೂಸುಫ್ ಮತ್ತು ಅವನ ತಮ್ಮನ ವಿಷಯದಲ್ಲಿ ಹೇಗೆ ವರ್ತಿಸಿದ್ದಿರೆಂದು ನಿಮಗೆ ತಿಳಿದಿದೆಯೇ?" (ಅವರು ಬೆರಗಾಗಿ ಕೇಳಿದರು) "ಏನು? ನೀವು ಯೂಸುಫರೇ?" (ಆಗ) ಅವರು, "ಹೌದು, ನಾನು ಯೂಸುಫ್ ಮತ್ತು ಇವನು ನನ್ನ ತಮ್ಮ. ಅಲ್ಲಾಹನು ನಮ್ಮ ಮೇಲೆ ಅನುಗ್ರಹ ತೋರಿದ್ದಾನೆ. ವಾಸ್ತವದಲ್ಲಿ ಭಯಭಕ್ತಿ ಮತ್ತು ಸಹನೆಯಿಂದ ಕಾರ್ಯವೆಸಗಿದರೆ ಅಲ್ಲಾಹನ ಬಳಿ ಇಂತಹ ಸಜ್ಜನರ ಪ್ರತಿಫಲ ವ್ಯರ್ಥಗೊಳ್ಳುವುದಿಲ್ಲ" ಎಂದರು. ಅವರು (ಹೀಗೆ) ಹೇಳಿದರು, "ಅಲ್ಲಾಹನಾಣೆ! ನಿಮಗೆ ಅಲ್ಲಾಹನು ನಮ್ಮ ಮೇಲೆ ಶ್ರೇಷ್ಠತೆ ದಯಪಾಲಿಸಿದ್ದಾನೆ. ನಿಜಕ್ಕೂ ನಾವು ಅಪರಾಧಿಗಳಾಗಿದ್ದೆವು." ಆಗ ಅವರು, "ಇಂದು ನಿಮ್ಮ ಮೇಲೆ ಯಾವ ದೋಷಾರೋಪಣೆಯೂ ಇಲ್ಲ. ಅಲ್ಲಾಹನು ನಿಮ್ಮನ್ನು ಕ್ಷಮಿಸಲಿ. ಅವನು ಎಲ್ಲರಿಗಿಂತಲೂ ಮಿಗಿಲಾದ ದಯಾಮಯನಾಗಿದ್ದಾನೆ. ಹೋಗಿರಿ, ನನ್ನ ಈ ಅಂಗಿಯನ್ನು ಕೊಂಡು ಹೋಗಿರಿ ಮತ್ತು ನನ್ನ ತಂದೆಯವರ ಮುಖದ ಮೇಲೆ ಹಾಕಿಬಿಡಿರಿ. ಅವರ ದೃಷ್ಟಿ ಮರಳಿ ಬರುವುದು ಮತ್ತು ನಿಮ್ಮ ಕುಟುಂಬದವರನ್ನೆಲ್ಲ ನನ್ನ ಬಳಿಗೆ ಕರೆತನ್ನಿರಿ" ಎಂದರು.
12:94 - 95
وَلَمَّا فَصَلَتِ ٱلْعِيرُ قَالَ أَبُوهُمْ إِنِّى لَأَجِدُ رِيحَ يُوسُفَ ۖ لَوْلَآ أَن تُفَنِّدُونِ٤٩قَالُوا۟ تَٱللَّهِ إِنَّكَ لَفِى ضَلَـٰلِكَ ٱلْقَدِيمِ٥٩
ಈ ಕಾರವಾನವು (ಈಜಿಪ್ಟಿನಿಂದ ಹೊರಟಾಗ) ಅವರ ತಂದೆಯು (ಕನ್‌ಆನ್‌ನಲ್ಲಿ ಹೀಗೆ) ಹೇಳಿದರು: "ನಾನು ಯೂಸುಫನ ಸುವಾಸನೆಯನ್ನು ಪಡೆಯುತ್ತಿದ್ದೇನೆ.66 ನೀವು ನನಗೆ ಮುದಿಭ್ರಾಂತಿಯುಂಟಾಗಿದೆ ಎಂದೇನೂ ಹೇಳಬೇಡಿರಿ." ಆಗ ಮನೆಯವರು, "ಅಲ್ಲಾಹನಾಣೆ ನೀವಿನ್ನೂ ಅದೇ ಹಳೆಯ ಭ್ರಮೆಯಲ್ಲೇ ಬಿದ್ದಿರುವಿರಿ" ಎಂದರು.67
12:96 - 98
فَلَمَّآ أَن جَآءَ ٱلْبَشِيرُ أَلْقَىٰهُ عَلَىٰ وَجْهِهِۦ فَٱرْتَدَّ بَصِيرًا ۖ قَالَ أَلَمْ أَقُل لَّكُمْ إِنِّىٓ أَعْلَمُ مِنَ ٱللَّهِ مَا لَا تَعْلَمُونَ٦٩قَالُوا۟ يَـٰٓأَبَانَا ٱسْتَغْفِرْ لَنَا ذُنُوبَنَآ إِنَّا كُنَّا خَـٰطِـِٔينَ٧٩قَالَ سَوْفَ أَسْتَغْفِرُ لَكُمْ رَبِّىٓ ۖ إِنَّهُۥ هُوَ ٱلْغَفُورُ ٱلرَّحِيمُ٨٩
ತರುವಾಯ ಸುವಾರ್ತೆ ತರುವವನು ಬಂದಾಗ ಅವನು ಯೂಸುಫರ ಅಂಗಿಯನ್ನು ಯಾ‌ಕೂಬರ ಮುಖದ ಮೇಲೆ ಹಾಕಿದನು ಮತ್ತು ಪಕ್ಕನೆ ಅವರ ದೃಷ್ಟಿ ಮರಳಿ ಬಂತು. ಆಗ ಅವರು ಹೇಳಿದರು, "ನಾನು ಮೊದಲೇ ಹೇಳುತ್ತಿರಲಿಲ್ಲವೇ? ನಾನು ಅಲ್ಲಾಹನ ಕಡೆಯಿಂದ ನಿಮಗೆ ತಿಳಿಯದುದನ್ನು ತಿಳಿಯುತ್ತೇನೆ." ಆಗ ಎಲ್ಲರೂ ಒಕ್ಕೊರಳಿನಿಂದ, "ಅಪ್ಪಾ, ನೀವು ನಮ್ಮ ಅಪರಾಧಗಳಿಗಾಗಿ ನಮ್ಮನ್ನು ಕ್ಷಮಿಸುವಂತೆ ಪ್ರಾರ್ಥಿಸಿರಿ. ನಿಜಕ್ಕೂ ನಾವು ಅಪರಾಧಿಗಳಾಗಿದ್ದೆವು" ಎಂದರು. ಆಗ ಅವರು, "ನಾನು ನನ್ನ ಪ್ರಭುವಿನೊಡನೆ ನಿಮ್ಮನ್ನು ಕ್ಷಮಿಸಲಿಕ್ಕಾಗಿ ಪ್ರಾರ್ಥಿಸುವೆನು. ಅವನು ಮಹಾ ಕ್ಷಮಾಶೀಲನೂ ಕರುಣಾನಿಧಿಯೂ ಆಗಿದ್ದಾನೆ" ಎಂದರು.
12:99 - 99
فَلَمَّا دَخَلُوا۟ عَلَىٰ يُوسُفَ ءَاوَىٰٓ إِلَيْهِ أَبَوَيْهِ وَقَالَ ٱدْخُلُوا۟ مِصْرَ إِن شَآءَ ٱللَّهُ ءَامِنِينَ٩٩
ಅನಂತರ ಇವರು ಯೂಸುಫರ ಬಳಿಗೆ ತಲಪಿದಾಗ68 ಅವರು ತಮ್ಮ ಮಾತಾಪಿತರನ್ನು ತನ್ನೊಂದಿಗೆ ಇರಿಸಿಕೊಂಡರು69 ಮತ್ತು (ತಮ್ಮ ಎಲ್ಲಾ ಕುಟುಂಬದವರೊಡನೆ) "ನೀವೆಲ್ಲರೂ ನಗರದೊಳಕ್ಕೆ ಪ್ರವೇಶಿಸಿರಿ. ಅಲ್ಲಾಹನಿಚ್ಛಿಸಿದರೆ ಇಲ್ಲಿ ಶಾಂತಿ ಸಮಾಧಾನಗಳೊಂದಿಗೆ ವಾಸಿಸುವಿರಿ." ಎಂದರು.
12:100 - 101
وَرَفَعَ أَبَوَيْهِ عَلَى ٱلْعَرْشِ وَخَرُّوا۟ لَهُۥ سُجَّدًا ۖ وَقَالَ يَـٰٓأَبَتِ هَـٰذَا تَأْوِيلُ رُءْيَـٰىَ مِن قَبْلُ قَدْ جَعَلَهَا رَبِّى حَقًّا ۖ وَقَدْ أَحْسَنَ بِىٓ إِذْ أَخْرَجَنِى مِنَ ٱلسِّجْنِ وَجَآءَ بِكُم مِّنَ ٱلْبَدْوِ مِنۢ بَعْدِ أَن نَّزَغَ ٱلشَّيْطَـٰنُ بَيْنِى وَبَيْنَ إِخْوَتِىٓ ۚ إِنَّ رَبِّى لَطِيفٌ لِّمَا يَشَآءُ ۚ إِنَّهُۥ هُوَ ٱلْعَلِيمُ ٱلْحَكِيمُ٠٠١۞ رَبِّ قَدْ ءَاتَيْتَنِى مِنَ ٱلْمُلْكِ وَعَلَّمْتَنِى مِن تَأْوِيلِ ٱلْأَحَادِيثِ ۚ فَاطِرَ ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ أَنتَ وَلِىِّۦ فِى ٱلدُّنْيَا وَٱلْـَٔاخِرَةِ ۖ تَوَفَّنِى مُسْلِمًا وَأَلْحِقْنِى بِٱلصَّـٰلِحِينَ١٠١
(ನಗರದೊಳಗೆ ಪ್ರವೇಶಿಸಿದ ತರುವಾಯ) ಅವರು ತಮ್ಮ ಮಾತಾಪಿತರನ್ನು ತಮ್ಮ ಬಳಿ ಆಸನದಲ್ಲಿ ಕುಳ್ಳಿರಿಸಿಕೊಂಡರು ಮತ್ತು ಎಲ್ಲರೂ ಅವರ ಮುಂದೆ ಸಜ್ದಃ ಮಾಡಿದರು.70 ಆಗ ಯೂಸುಫ್ ಹೀಗೆಂದರು: "ಅಪ್ಪಾ, ನಾನು ಹಿಂದೆ ಕಂಡಿದ್ದ ಸ್ವಪ್ನದ ಫಲವಿದು. ನನ್ನ ಪ್ರಭು ಅದನ್ನು ಸಾಕಾರಗೊಳಿಸಿದನು. ನನ್ನನ್ನು ಸೆರೆಮನೆಯಿಂದ ಬಿಡಿಸಿ ತಮ್ಮನ್ನೆಲ್ಲಾ ಮರುಭೂಮಿಯಿಂದ ತಂದು ನನ್ನೊಂದಿಗೆ ಸೇರಿಸಿದುದು ಅವನ ಮಹದುಪಕಾರವಾಗಿದೆ. ವಸ್ತುತಃ ಶೈತಾನನು ನನ್ನ ಮತ್ತು ನನ್ನ ಸಹೋದರರ ನಡುವೆ ಪಿತೂರಿಯನ್ನು ಮಾಡಿದ್ದನು. ವಾಸ್ತವದಲ್ಲಿ ನನ್ನ ಪ್ರಭು ಗ್ರಹಿಸಲಾಗದಂತಹ ಯುಕ್ತಿಗಳಿಂದ ತನ್ನ ವಿಧಿಯನ್ನು ಪೂರ್ತೀಕರಿಸುತ್ತಾನೆ. ನಿಶ್ಚಯವಾಗಿಯೂ ಅವನು ಸರ್ವಜ್ಞನೂ ಧೀಮಂತನೂ ಆಗಿರುತ್ತಾನೆ. ಓ ನನ್ನ ಪ್ರಭೂ, ನೀನು ನನಗೆ ಸಾಮ್ರಾಜ್ಯಾಧಿಕಾರವನ್ನು ದಯಪಾಲಿಸಿದೆ. ವಿಷಯಗಳ ಆಳವನ್ನರಿಯಲು ಕಲಿಸಿಕೊಟ್ಟೆ. ಓ ಆಕಾಶಗಳ ಮತ್ತು ಭೂಮಿಯ ನಿರ್ಮಾಪಕನೇ! ಇಹ ಮತ್ತು ಪರದಲ್ಲಿ ನೀನೇ ನನ್ನ ರಕ್ಷಕ ಮಿತ್ರನು. ನನ್ನನ್ನು ಮುಸ್ಲಿಮನನ್ನಾಗಿ ಅಂತ್ಯಗೊಳಿಸು ಮತ್ತು ಅಂತಿಮವಾಗಿ ಸಜ್ಜನರೊಂದಿಗೆ ಸೇರಿಸು.71
12:102 - 104
ذَٰلِكَ مِنْ أَنۢبَآءِ ٱلْغَيْبِ نُوحِيهِ إِلَيْكَ ۖ وَمَا كُنتَ لَدَيْهِمْ إِذْ أَجْمَعُوٓا۟ أَمْرَهُمْ وَهُمْ يَمْكُرُونَ٢٠١وَمَآ أَكْثَرُ ٱلنَّاسِ وَلَوْ حَرَصْتَ بِمُؤْمِنِينَ٣٠١وَمَا تَسْـَٔلُهُمْ عَلَيْهِ مِنْ أَجْرٍ ۚ إِنْ هُوَ إِلَّا ذِكْرٌ لِّلْعَـٰلَمِينَ٤٠١
ಓ ಪೈಗಂಬರರೇ, ನಿಮ್ಮ ಮೇಲೆ ಅವತೀರ್ಣಗೊಳಿಸುತ್ತಿರುವ ಈ ಚರಿತ್ರೆಯು ಅದೃಶ್ಯ ವೃತ್ತಾಂತಗಳಲ್ಲಾಗಿದೆ. ಅನ್ಯಥಾ ಯೂಸುಫರ ಸಹೋದರರು ಪರಸ್ಪರ ಒಮ್ಮತದೊಂದಿಗೆ ಕುತಂತ್ರ ಹೂಡಿದಾಗ ನೀವು ಅಲ್ಲಿರಲಿಲ್ಲ. ಆದರೆ ನೀವು ಎಷ್ಟೇ ಆಗ್ರಹಿಸಿದರೂ ಇವರಲ್ಲಿ ಹೆಚ್ಚಿನವರು ವಿಶ್ವಾಸ ಹೊಂದುವವರಲ್ಲ.72 ವಸ್ತುತಃ ನೀವು ಅವರಿಂದ ಈ ಸೇವೆಗೆ ಯಾವ ಪ್ರತಿಫಲವನ್ನೂ ಬೇಡುವುದಿಲ್ಲ. ಇದು ಜಗತ್ತಿನವರಿಗೊಂದು ಸಾರ್ವತ್ರಿಕ ಉಪದೇಶ.73
12:105 - 108
وَكَأَيِّن مِّنْ ءَايَةٍ فِى ٱلسَّمَـٰوَٰتِ وَٱلْأَرْضِ يَمُرُّونَ عَلَيْهَا وَهُمْ عَنْهَا مُعْرِضُونَ٥٠١وَمَا يُؤْمِنُ أَكْثَرُهُم بِٱللَّهِ إِلَّا وَهُم مُّشْرِكُونَ٦٠١أَفَأَمِنُوٓا۟ أَن تَأْتِيَهُمْ غَـٰشِيَةٌ مِّنْ عَذَابِ ٱللَّهِ أَوْ تَأْتِيَهُمُ ٱلسَّاعَةُ بَغْتَةً وَهُمْ لَا يَشْعُرُونَ٧٠١قُلْ هَـٰذِهِۦ سَبِيلِىٓ أَدْعُوٓا۟ إِلَى ٱللَّهِ ۚ عَلَىٰ بَصِيرَةٍ أَنَا۠ وَمَنِ ٱتَّبَعَنِى ۖ وَسُبْحَـٰنَ ٱللَّهِ وَمَآ أَنَا۠ مِنَ ٱلْمُشْرِكِينَ٨٠١
ಆಕಾಶಗಳಲ್ಲೂ74 ಭೂಮಿಯಲ್ಲೂ ಎಷ್ಟೋ ನಿದರ್ಶನಗಳಿವೆ. ಅವುಗಳನ್ನು ಇವರು ಹಾದುಕೊಂಡು ಹೋಗುತ್ತಿರುತ್ತಾರೆ ಮತ್ತು ಅವುಗಳ ಕಡೆಗೆ ಕಿಂಚಿತ್ ಗಮನಹರಿಸುವುದಿಲ್ಲ.75 ಇವರಲ್ಲಿ ಹೆಚ್ಚಿನವರು ಅಲ್ಲಾಹನನ್ನು ನಂಬುತ್ತಾರೆ. ಆದರೆ ಅವನೊಂದಿಗೆ ಇತರರನ್ನು ಸಹಭಾಗಿಗಳನ್ನಾಗಿಸಿಕೊಂಡು (ನಂಬುತ್ತಾರೆ).76 ಅಲ್ಲಾಹನ ಯಾತನೆಯ ಒಂದಂಶ ವಿಪತ್ತು ಥಟ್ಟನೆ ಇವರ ಮೇಲೆ ಬೀಳಲಾರದೆಂದು ಅಥವಾ ಲೋಕಾಂತ್ಯವಾಗುವ ಆ ಘಳಿಗೆಯು ಹಠಾತ್ತನೆ ಅವರ ಮೇಲೆ ಬಂದೆರಗಲಾರದೆಂದು ಇವರೇನು ನಿಶ್ಚಿಂತರಾಗಿರುವರೇ?77 ನೀವು ಇವರೊಡನೆ ಸ್ಪಷ್ಟವಾಗಿ ಹೇಳಿಬಿಡಿರಿ - "ಇದೇ ನನ್ನ ಮಾರ್ಗ. ನಾನು ಅಲ್ಲಾಹನ ಕಡೆಗೆ ಕರೆ ನೀಡುತ್ತೇನೆ. ನಾನೂ ನನ್ನ ಸಂಗಾತಿಗಳೂ ಸಂಪೂರ್ಣ ಪ್ರಕಾಶದಲ್ಲಿ ದಾರಿಗಾಣುತ್ತಿದ್ದೇವೆ. ಅಲ್ಲಾಹನು ಅತ್ಯಂತ ಪಾವನನು.78 ಸಹಭಾಗಿಗಳಾಗಿಸುವವರೊಂದಿಗೆ ನನಗೆ ಯಾವ ಸಂಬಂಧವೂ ಇಲ್ಲ."
12:109 - 110
وَمَآ أَرْسَلْنَا مِن قَبْلِكَ إِلَّا رِجَالًا نُّوحِىٓ إِلَيْهِم مِّنْ أَهْلِ ٱلْقُرَىٰٓ ۗ أَفَلَمْ يَسِيرُوا۟ فِى ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَـٰقِبَةُ ٱلَّذِينَ مِن قَبْلِهِمْ ۗ وَلَدَارُ ٱلْـَٔاخِرَةِ خَيْرٌ لِّلَّذِينَ ٱتَّقَوْا۟ ۗ أَفَلَا تَعْقِلُونَ٩٠١حَتَّىٰٓ إِذَا ٱسْتَيْـَٔسَ ٱلرُّسُلُ وَظَنُّوٓا۟ أَنَّهُمْ قَدْ كُذِبُوا۟ جَآءَهُمْ نَصْرُنَا فَنُجِّىَ مَن نَّشَآءُ ۖ وَلَا يُرَدُّ بَأْسُنَا عَنِ ٱلْقَوْمِ ٱلْمُجْرِمِينَ٠١١
ಓ ಪೈಗಂಬರರೇ, ನಿಮಗಿಂತ ಮುಂಚೆ ನಾವು ಕಳುಹಿಸಿದ್ದ ಪ್ರವಾದಿಗಳೆಲ್ಲರೂ ಪುರುಷರೇ ಆಗಿದ್ದರು. ಆಯಾ ನಾಡುಗಳಲ್ಲೇ ವಾಸಿಸುವವರಾಗಿದ್ದರು ಮತ್ತು ಅವರ ಕಡೆಗೆ ನಾವು ‘ದಿವ್ಯವಾಣಿ’ಯನ್ನು ಕಳುಹಿಸುತ್ತಲಿದ್ದೆವು. ಇವರಿಗಿಂತ ಮುಂಚಿನ ಜನಾಂಗಗಳ ಪರಿಣಾಮವು ಇವರಿಗೆ ಕಾಣದಿರಬೇಕಾದರೆ ಇವರು ಭೂಮಿಯ ಮೇಲೆ ಸಂಚರಿಸಲಿಲ್ಲವೇ? (ಪ್ರವಾದಿಗಳ ಬೋಧನೆಯನ್ನು ಸ್ವೀಕರಿಸಿ) ದೇವಭಯವಿರಿಸಿಕೊಂಡವರಿಗೆ ನಿಶ್ಚಯವಾಗಿಯೂ ಪರಲೋಕ ಗೇಹವು ಇನ್ನಷ್ಟು ಉತ್ತಮ. ಇನ್ನಾದರೂ ನೀವು ವಿವೇಚಿಸಲಾರಿರಾ?79 (ಗತ ಪ್ರವಾದಿಗಳೂ ದೀರ್ಘಕಾಲದ ವರೆಗೆ ಬೋಧನೆ ನೀಡುತ್ತಲೇ ಇದ್ದರೂ ಜನರು ಕಿವಿ ಕೊಡಲಿಲ್ಲ) ಕೊನೆಗೆ ಪ್ರವಾದಿಗಳು ಜನರ ಬಗ್ಗೆ ನಿರಾಶೆಗೊಂಡಾಗ ಹಾಗೂ ಅವರೊಡನೆ ಸುಳ್ಳೇ ಹೇಳಲಾಗಿತ್ತೆಂದು ಜನರೂ ತಿಳಿದುಕೊಂಡಾಗ ಹಠಾತ್ತನೆ ನಮ್ಮ ಸಹಾಯವು ಪ್ರವಾದಿಗಳಿಗೆ ತಲಪಿತು. ಇಂತಹ ಸಂದರ್ಭ ಬಂದಾಗ ನಮಗಿಷ್ಟ ಬಂದವರನ್ನು ರಕ್ಷಿಸಿಕೊಳ್ಳುವುದು ನಮ್ಮ ರೂಢಿ. ಅಪರಾಧಿಗಳ ಮೇಲಿಂದ ನಮ್ಮ ಯಾತನೆಯು ಎಷ್ಟಕ್ಕೂ ಸರಿಯಲಾರದು.
12:111 - 111
لَقَدْ كَانَ فِى قَصَصِهِمْ عِبْرَةٌ لِّأُو۟لِى ٱلْأَلْبَـٰبِ ۗ مَا كَانَ حَدِيثًا يُفْتَرَىٰ وَلَـٰكِن تَصْدِيقَ ٱلَّذِى بَيْنَ يَدَيْهِ وَتَفْصِيلَ كُلِّ شَىْءٍ وَهُدًى وَرَحْمَةً لِّقَوْمٍ يُؤْمِنُونَ١١١
ಗತ ಸಮುದಾಯಗಳ ಈ ಕಥೆಗಳಲ್ಲಿ ಬುದ್ಧಿಜೀವಿಗಳಿಗೆ ಪಾಠವಿದೆ. ಕುರ್‌ಆನಿನಲ್ಲಿ ವಿವರಿಸಲ್ಪಡುತ್ತಿರುವುದೆಲ್ಲ ಕೃತಕ ವಿಷಯಗಳಲ್ಲ. ಬದಲಾಗಿ ಇದಕ್ಕಿಂತ ಮುಂಚೆ ಬಂದಿರುವ ಗ್ರಂಥಗಳ ದೃಢೀಕರಣವೂ ಪ್ರತಿಯೊಂದು ವಿಷಯದ ವಿವರಣೆಯೂ80 ಸತ್ಯವಿಶ್ವಾಸವನ್ನು ಸ್ವೀಕರಿಸಿಕೊಳ್ಳುವವರಿಗೆ ಸನ್ಮಾರ್ಗದರ್ಶನವೂ ಅನುಗ್ರಹವೂ ಆಗಿರುತ್ತದೆ.
