72. ಅಲ್ ಜಿನ್ನ್
ಪರಮ ದಯಾಮಯನೂ ಕರುಣಾಳುವೂ ಆದ ಅಲ್ಲಾಹನ ನಾಮದಿಂದ
72:1 - 3
بِسْمِ ٱللَّهِ ٱلرَّحْمَـٰنِ ٱلرَّحِيمِ قُلْ أُوحِىَ إِلَىَّ أَنَّهُ ٱسْتَمَعَ نَفَرٌ مِّنَ ٱلْجِنِّ فَقَالُوٓا۟ إِنَّا سَمِعْنَا قُرْءَانًا عَجَبًا١يَهْدِىٓ إِلَى ٱلرُّشْدِ فَـَٔامَنَّا بِهِۦ ۖ وَلَن نُّشْرِكَ بِرَبِّنَآ أَحَدًا٢وَأَنَّهُۥ تَعَـٰلَىٰ جَدُّ رَبِّنَا مَا ٱتَّخَذَ صَـٰحِبَةً وَلَا وَلَدًا٣
ಕಡೆಗೆ ಮಾರ್ಗದರ್ಶನ ಮಾಡುತ್ತದೆ. ಆದ್ದರಿಂದ ನಾವು ಅದರ ಮೇಲೆ ವಿಶ್ವಾಸವಿರಿಸಿದೆವು ಮತ್ತು ನಾವಿನ್ನು ಎಷ್ಟು ಮಾತ್ರಕ್ಕೂ ನಮ್ಮ ಪ್ರಭುವಿನೊಂದಿಗೆ ಯಾರನ್ನೂ ಸಹಭಾಗಿಯಾಗಿ ಮಾಡಲಾರೆವು.3 ಅಲ್ಲದೆ, `ನಮ್ಮ ಪ್ರಭುವಿನ ಪ್ರತಿಷ್ಠೆಯು ಅತ್ಯುನ್ನತವಾಗಿದೆ. ಅವನು ಯಾರನ್ನೂ ತನ್ನ ಪತ್ನಿ ಅಥವಾ ಪುತ್ರನಾಗಿ ಮಾಡಿಕೊಂಡಿಲ್ಲ.'4
72:4 - 4
وَأَنَّهُۥ كَانَ يَقُولُ سَفِيهُنَا عَلَى ٱللَّهِ شَطَطًا٤
ಅಲ್ಲದೆ, `ನಮ್ಮಲ್ಲಿನ ತಿಳಿಗೇಡಿಗಳು5 ಅಲ್ಲಾಹನ ವಿಷಯದಲ್ಲಿ ಘೋರ ಅಸತ್ಯವನ್ನು ಹೇಳುತ್ತಿರುತ್ತಾರೆ.'
72:5 - 5
وَأَنَّا ظَنَنَّآ أَن لَّن تَقُولَ ٱلْإِنسُ وَٱلْجِنُّ عَلَى ٱللَّهِ كَذِبًا٥
ಅಲ್ಲದೆ, `ಮಾನವರು ಮತ್ತು ಜಿನ್ನ್ ಗಳು ಅಲ್ಲಾಹನ ವಿಷಯದಲ್ಲಿ ಎಂದೂ ಸುಳ್ಳು ಹೇಳಲಾರರೆಂದು ನಾವು ಭಾವಿಸಿಕೊಂಡಿದ್ದೆವು.'6
72:6 - 15
وَأَنَّهُۥ كَانَ رِجَالٌ مِّنَ ٱلْإِنسِ يَعُوذُونَ بِرِجَالٍ مِّنَ ٱلْجِنِّ فَزَادُوهُمْ رَهَقًا٦وَأَنَّهُمْ ظَنُّوا۟ كَمَا ظَنَنتُمْ أَن لَّن يَبْعَثَ ٱللَّهُ أَحَدًا٧وَأَنَّا لَمَسْنَا ٱلسَّمَآءَ فَوَجَدْنَـٰهَا مُلِئَتْ حَرَسًا شَدِيدًا وَشُهُبًا٨وَأَنَّا كُنَّا نَقْعُدُ مِنْهَا مَقَـٰعِدَ لِلسَّمْعِ ۖ فَمَن يَسْتَمِعِ ٱلْـَٔانَ يَجِدْ لَهُۥ شِهَابًا رَّصَدًا٩وَأَنَّا لَا نَدْرِىٓ أَشَرٌّ أُرِيدَ بِمَن فِى ٱلْأَرْضِ أَمْ أَرَادَ بِهِمْ رَبُّهُمْ رَشَدًا٠١وَأَنَّا مِنَّا ٱلصَّـٰلِحُونَ وَمِنَّا دُونَ ذَٰلِكَ ۖ كُنَّا طَرَآئِقَ قِدَدًا١١وَأَنَّا ظَنَنَّآ أَن لَّن نُّعْجِزَ ٱللَّهَ فِى ٱلْأَرْضِ وَلَن نُّعْجِزَهُۥ هَرَبًا٢١وَأَنَّا لَمَّا سَمِعْنَا ٱلْهُدَىٰٓ ءَامَنَّا بِهِۦ ۖ فَمَن يُؤْمِنۢ بِرَبِّهِۦ فَلَا يَخَافُ بَخْسًا وَلَا رَهَقًا٣١وَأَنَّا مِنَّا ٱلْمُسْلِمُونَ وَمِنَّا ٱلْقَـٰسِطُونَ ۖ فَمَنْ أَسْلَمَ فَأُو۟لَـٰٓئِكَ تَحَرَّوْا۟ رَشَدًا٤١وَأَمَّا ٱلْقَـٰسِطُونَ فَكَانُوا۟ لِجَهَنَّمَ حَطَبًا٥١
ಅಲ್ಲದೆ, `ಮಾನವರ ಪೈಕಿ ಕೆಲವರು ಜಿನ್ನ್ ಗಳ ಪೈಕಿ ಕೆಲವರ ಅಭಯ ಯಾಚಿಸುತ್ತಿದ್ದರು. ಹೀಗೆ ಅವರು ಜಿನ್ನ್ ಗಳ ಅಹಂಕಾರವನ್ನು ಇನ್ನಷ್ಟು ಹೆಚ್ಚಿಸಿದರು.'7 ಅಲ್ಲದೆ `ಮಾನವರು ಕೂಡಾ ನಿಮ್ಮಂತೆಯೇ, ಅಲ್ಲಾಹನು ಯಾರನ್ನೂ ಸಂದೇಶವಾಹಕನಾಗಿ ಕಳುಹಿಸುವುದಿಲ್ಲವೆಂದು ಭಾವಿಸಿಕೊಂಡಿದ್ದರು.8 ಅಲ್ಲದೆ, `ನಾವು ಆಕಾಶವನ್ನು ಹುಡುಕಾಡಿದಾಗ, ಅದು ಕಾವಲುಗಾರರಿಂದ ತುಂಬಿರುವುದನ್ನು ಮತ್ತು ಉಲ್ಕೆಗಳ ಮಳೆ ಸುರಿಯುತ್ತಿರುವುದನ್ನು ಕಂಡೆವು.` ಅಲ್ಲದೆ, `ಹಿಂದೆ ನಾವು ಕದ್ದಾಲಿಸಲಿಕ್ಕಾಗಿ ಆಕಾಶದಲ್ಲಿ ಕೂರಲು ಸ್ಥಳ ಪಡೆಯುತ್ತಿದ್ದೆವು. ಆದರೆ ಈಗ ಕದ್ದಾಲಿಸಲು ಪ್ರಯತ್ನಿಸುವವನು ತನಗಾಗಿ ಒಂದು ಉಲ್ಕೆಯು ಹೊಂಚು ಹಾಕುತ್ತಿರುವುದನ್ನು ಕಾಣುತ್ತಾನೆ.'9 ಅಲ್ಲದೆ `ಭೂಮಿಯ ನಿವಾಸಿಗಳೊಂದಿಗೆ ಏನಾದರೂ ಕೆಟ್ಟ ವ್ಯವಹಾರ ಮಾಡಲು ನಿರ್ಧರಿಸಲಾಗಿದೆಯೋ ಅಥವಾ ಅವರ ಪ್ರಭುವು ಅವರಿಗೆ ಸನ್ಮಾರ್ಗವನ್ನು ತೋರಿಸ ಬಯಸಿರುವನೋ ಎಂಬುದು ನಮಗೆ ಅರ್ಥವಾಗುತ್ತಿರಲಿಲ್ಲ.'10 ಅಲ್ಲದೆ, `ನಮ್ಮಲ್ಲಿ ಕೆಲವರು ಸಜ್ಜನರಿದ್ದಾರೆ ಮತ್ತೆ ಕೆಲವರು ಅದಕ್ಕಿಂತ ಕೆಳಮಟ್ಟದವರಾಗಿದ್ದಾರೆ. ನಾವು ನಾನಾ ಪದ್ಧತಿಗಳಲ್ಲಿ ಹರಿಹಂಚಿ ಹೋಗಿದ್ದೇವೆ.11 ಅಲ್ಲದೆ, ನಾವು ಭೂಮಿಯಲ್ಲಿಯೂ ಅಲ್ಲಾಹನನ್ನು ಸೋಲಿಸಲಾರೆವು ಮತ್ತು ಓಡಿ ಹೋಗಿಯೂ ಅವನನ್ನು ಸೋಲಿಸಲಾರೆವು ಎಂದು ನಾವು ತಿಳಿದುಕೊಂಡಿದ್ದೆವು.'12 ಅಲ್ಲದೆ, `ನಾವು ಸನ್ಮಾರ್ಗದ ಬೋಧನೆಯನ್ನು ಕೇಳಿದಾಗ ಅದರ ಮೇಲೆ ವಿಶ್ವಾಸವಿರಿಸಿದೆವು. ಇನ್ನು ತನ್ನ ಪ್ರಭುವಿನ ಮೇಲೆ ಯಾರು ವಿಶ್ವಾಸವಿರಿಸುವನೋ ಅವನಿಗೆ ಯಾವುದೇ ಹಕ್ಕುಚ್ಯುತಿ ಅಥವಾ ಅನ್ಯಾಯದ ಭಯವಿರಲಾರದು.'13 ಅಲ್ಲದೆ, ನಮ್ಮಲ್ಲಿ ಕೆಲವು ಮುಸ್ಲಿಮರು (ಅಲ್ಲಾಹನಿಗೆ ವಿಧೇಯರಾದವರು) ಇದ್ದಾರೆ, ಇನ್ನು ಕೆಲವರು ಸತ್ಯದಿಂದ ವಿಮುಖರಾಗಿದ್ದಾರೆ. ಇಸ್ಲಾಮನ್ನು (ಆಜ್ಞಾಪಾಲನೆಯ ಮಾರ್ಗವನ್ನು) ಸ್ವೀಕರಿಸಿಕೊಂಡವರು ಮೋಕ್ಷ ಮಾರ್ಗವನ್ನು ಕಂಡುಕೊಂಡರು ಮತ್ತು ಸತ್ಯದಿಂದ ವಿಮುಖರಾದವರು ನರಕದ ಇಂಧನವಾಗುವರು.14
72:16 - 23
وَأَلَّوِ ٱسْتَقَـٰمُوا۟ عَلَى ٱلطَّرِيقَةِ لَأَسْقَيْنَـٰهُم مَّآءً غَدَقًا٦١لِّنَفْتِنَهُمْ فِيهِ ۚ وَمَن يُعْرِضْ عَن ذِكْرِ رَبِّهِۦ يَسْلُكْهُ عَذَابًا صَعَدًا٧١وَأَنَّ ٱلْمَسَـٰجِدَ لِلَّهِ فَلَا تَدْعُوا۟ مَعَ ٱللَّهِ أَحَدًا٨١وَأَنَّهُۥ لَمَّا قَامَ عَبْدُ ٱللَّهِ يَدْعُوهُ كَادُوا۟ يَكُونُونَ عَلَيْهِ لِبَدًا٩١قُلْ إِنَّمَآ أَدْعُوا۟ رَبِّى وَلَآ أُشْرِكُ بِهِۦٓ أَحَدًا٠٢قُلْ إِنِّى لَآ أَمْلِكُ لَكُمْ ضَرًّا وَلَا رَشَدًا١٢قُلْ إِنِّى لَن يُجِيرَنِى مِنَ ٱللَّهِ أَحَدٌ وَلَنْ أَجِدَ مِن دُونِهِۦ مُلْتَحَدًا٢٢إِلَّا بَلَـٰغًا مِّنَ ٱللَّهِ وَرِسَـٰلَـٰتِهِۦ ۚ وَمَن يَعْصِ ٱللَّهَ وَرَسُولَهُۥ فَإِنَّ لَهُۥ نَارَ جَهَنَّمَ خَـٰلِدِينَ فِيهَآ أَبَدًا٣٢
(ಸಂದೇಶವಾಹಕರೇ, ನನಗೆ ಹೀಗೆಯೂ ವಹ್ಯ್ ಮಾಡಲಾಗಿದೆಯೆಂದು ಹೇಳಿರಿ:)15 ಜನರು ಸನ್ಮಾರ್ಗದಲ್ಲಿ ಸ್ಥಿರಚಿತ್ತರಾಗಿ ನಡೆಯುತ್ತಿದ್ದರೆ, ನಾವು ಅವರನ್ನು ಜಲಸಮೃದ್ಧರನ್ನಾಗಿ ಮಾಡಿಬಿಡುತ್ತಿದ್ದೆವು.16 ಈ ಅನುಗ್ರಹದ ಮೂಲಕ ಅವರನ್ನು ಪರೀಕ್ಷಿಸಲಿಕ್ಕಾಗಿ.17 ಇನ್ನು ಯಾರು ತನ್ನ ಪ್ರಭುವಿನ ದಿಕ್ರ್ ನಿಂದ (ಸ್ಮರಣೆಯಿಂದ) ವಿಮುಖನಾಗುವನೋ,18 ಅವನನ್ನು ಅವನ ಪ್ರಭುವು ಕಠಿಣ ಯಾತನೆಗೆ ಗುರಿಪಡಿಸುವನು. ಮತ್ತು ಮಸೀದಿಗಳು ಅಲ್ಲಾಹನಿಗಾಗಿವೆ. ಆದುದರಿಂದ ಅವುಗಳಲ್ಲಿ ಅಲ್ಲಾಹನೊಂದಿಗೆ ಇತರ ಯಾರನ್ನೂ ಪ್ರಾರ್ಥಿಸಬೇಡಿರಿ.19 ಅಲ್ಲಾಹನ ದಾಸನು20 ಅವನನ್ನು ಪ್ರಾರ್ಥಿಸಲಿಕ್ಕಾಗಿ ನಿಂತಾಗ, ಜನರು ಅವನ ಮೇಲೆ ಮುಗಿಬೀಳಲು ಸಿದ್ಧರಾಗಿ ಬಿಟ್ಟರು. ಸಂದೇಶವಾಹಕರೇ, ಹೇಳಿರಿ `ನಾನಂತು ನನ್ನ ಪ್ರಭುವನ್ನು ಪ್ರಾರ್ಥಿಸುತ್ತೇನೆ ಮತ್ತು ಅವನೊಂದಿಗೆ ಯಾರನ್ನೂ ಸಹಭಾಗಿಯಾಗಿ ಮಾಡುವುದಿಲ್ಲ.'21 ಹೇಳಿರಿ `ನಿಮಗೆ ಯಾವುದೇ ಹಾನಿ ಮಾಡಲಿಕ್ಕಾಗಲಿ, ಯಾವುದೇ ಲಾಭವನ್ನುಂಟು ಮಾಡಲಿಕ್ಕಾಗಲಿ ನನಗೆ ಅಧಿಕಾರವಿಲ್ಲ.' ಹೇಳಿರಿ `ನನ್ನನ್ನು ಅಲ್ಲಾಹನ ಹಿಡಿತದಿಂದ ಯಾರೂ ರಕ್ಷಿಸಲಾರರು ಮತ್ತು ನನಗೆ ಅವನ ಸನ್ನಿಧಿಯ ಹೊರತು ಬೇರೆಲ್ಲೂ ಅಭಯಾಶ್ರಯವೂ ಸಿಗಲಾರದು. ಅಲ್ಲಾಹನ ವಚನ ಮತ್ತು ಅವನ ಸಂದೇಶಗಳನ್ನು ತಲುಪಿಸುವ ಹೊರತು ಬೇರಾವ ಕರ್ತವ್ಯವೂ ನನ್ನ ಮೇಲಿಲ್ಲ.22 ಅಲ್ಲಾಹ್ ಮತ್ತು ಅವನ ಸಂದೇಶವಾಹಕರ ಆಜ್ಞೆಗಳನ್ನು ಉಲ್ಲಂಘಿಸುವವರಿಗಾಗಿ ನರಕಾಗ್ನಿ ಇದೆ ಮತ್ತು ಇಂತಹವರು ಅದರಲ್ಲಿ ಸದಾಕಾಲ ಬಿದ್ದಿರುವರು.'23
72:24 - 28
حَتَّىٰٓ إِذَا رَأَوْا۟ مَا يُوعَدُونَ فَسَيَعْلَمُونَ مَنْ أَضْعَفُ نَاصِرًا وَأَقَلُّ عَدَدًا٤٢قُلْ إِنْ أَدْرِىٓ أَقَرِيبٌ مَّا تُوعَدُونَ أَمْ يَجْعَلُ لَهُۥ رَبِّىٓ أَمَدًا٥٢عَـٰلِمُ ٱلْغَيْبِ فَلَا يُظْهِرُ عَلَىٰ غَيْبِهِۦٓ أَحَدًا٦٢إِلَّا مَنِ ٱرْتَضَىٰ مِن رَّسُولٍ فَإِنَّهُۥ يَسْلُكُ مِنۢ بَيْنِ يَدَيْهِ وَمِنْ خَلْفِهِۦ رَصَدًا٧٢لِّيَعْلَمَ أَن قَدْ أَبْلَغُوا۟ رِسَـٰلَـٰتِ رَبِّهِمْ وَأَحَاطَ بِمَا لَدَيْهِمْ وَأَحْصَىٰ كُلَّ شَىْءٍ عَدَدًۢا٨٢
ಇವರಿಗೆ ವಾಗ್ದಾನ ಮಾಡಲಾಗುತ್ತಿರುವುದನ್ನು ಕಣ್ಣಾರೆ ಕಂಡುಕೊಳ್ಳುವವರೆಗೂ (ಇವರು ತಮ್ಮ ಈ ನಿಲುಮೆಯನ್ನು ಕೈ ಬಿಡಲಾರರು) ಆಗ, ಯಾರ ಸಹಾಯಕರು ದುರ್ಬಲರು ಮತ್ತು ಯಾರ ಕೂಟವು ಅಲ್ಪಸಂಖ್ಯೆಯಲ್ಲಿದೆ ಎಂಬುದು ಅವರಿಗೆ ತಿಳಿದು ಬಿಡುವುದು.24 ಹೇಳಿರಿ, `ನಿಮಗೆ ಏನನ್ನು ವಾಗ್ದಾನ ಮಾಡಲಾಗುತ್ತಿದೆಯೋ ಅದು ಸಮೀಪವಿದೆಯೋ ಅಥವಾ ನನ್ನ ಪ್ರಭು ಅದಕ್ಕಾಗಿ ದೀರ್ಘವಾದ ಅವಧಿಯನ್ನೇನಾದರೂ ನಿಶ್ಚಯಿಸುವನೋ ಎಂಬುದು ನನಗೆ ತಿಳಿದಿಲ್ಲ.25 ಅವನು ಅದೃಶ್ಯ ಜ್ಞಾನಿಯಾಗಿದ್ದಾನೆ. (ಅದೃಶ್ಯ ಜ್ಞಾನವನ್ನು ನೀಡಲಿಕ್ಕಾಗಿ) ತಾನು ಮೆಚ್ಚಿಕೊಂಡ, ತನ್ನ ಸಂದೇಶವಾಹಕನ ಹೊರತು26 ಬೇರಾರಿಗೂ ಅವನು ತನ್ನ ಅದೃಶ್ಯ ಜ್ಞಾನವನ್ನು ತಿಳಿಯಪಡಿಸುವುದಿಲ್ಲ.27 ಆಗ ಅವನ (ಸಂದೇಶವಾಹಕನ) ಮುಂದೆಯೂ ಹಿಂದೆಯೂ ಕಾವಲುಗಾರರನ್ನು ನೇಮಿಸಿ ಬಿಡುತ್ತಾನೆ.28 ಅವರು ತಮ್ಮ ಪ್ರಭುವಿನ ಸಂದೇಶಗಳನ್ನು ತಲುಪಿಸಿರುವರೆಂಬುದನ್ನು ಅವನು ತಿಳಿಯಲಿಕ್ಕಾಗಿ.29 ಅವನು ಅವರ ಪರಿಸರವನ್ನು ಸಂಪೂರ್ಣವಾಗಿ ಸುತ್ತುಗಟ್ಟಿರುತ್ತಾನೆ ಮತ್ತು ಪ್ರತಿಯೊಂದು ವಸ್ತುವನ್ನೂ ಅವನು ಲೆಕ್ಕ ಮಾಡಿಟ್ಟಿರುತ್ತಾನೆ.'30
